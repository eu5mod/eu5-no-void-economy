namespace = modeu5_us02_debug

modeu5_us02_debug.1 = {
	type = country_event
	title = modeu5_us02_debug.1.title
	desc = modeu5_us02_debug.1.desc
	outcome = neutral
	orphan = yes

	trigger = {
		always = no
	}

	option = {
		name = modeu5_us02_debug.1.a
		modeu5_debug_clear_stock_test_results = yes
		debug_log = "ModeU5 US-02 storage-capacity test starting."
		modeu5_debug_run_storage_capacity_test = yes
		debug_log = "ModeU5 US-02 storage-capacity test finished; inspect PASS/FAIL result event and modeu5_test_* markers."
		trigger_event_non_silently = modeu5_us02_debug.2
	}
	option = {
		name = modeu5_debug.close
	}
}

modeu5_us02_debug.2 = {
	type = country_event
	title = modeu5_us02_debug.2.title
	desc = modeu5_us02_debug.2.desc
	outcome = neutral

	trigger = {
		always = no
	}

	option = {
		name = modeu5_debug.test.storage_capacity.pass
		trigger = { has_global_variable = modeu5_test_storage_capacity_passed }
	}
	option = {
		name = modeu5_debug.test.storage_capacity.fail
		trigger = { NOT = { has_global_variable = modeu5_test_storage_capacity_passed } }
	}
	option = {
		name = modeu5_debug.close
	}
}
