# Deterministic US-02 manual-test helper.
# Recalculation may replace capacity maps but must never mutate stock.

modeu5_debug_run_storage_capacity_test = {
	save_temporary_scope_as = modeu5_capacity_test_controller

	if = {
		limit = { country_exists = c:FRA }
		c:FRA.capital.market = { save_temporary_scope_as = modeu5_capacity_test_market }

		modeu5_read_country_stock_record = {
			country = c:FRA
			market = scope:modeu5_capacity_test_market
			good = wheat
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_test_wheat_stock_before
			value = scope:modeu5_read_country_stock
		}

		modeu5_recalculate_country_storage_capacities = {
			country = c:FRA
		}

		modeu5_read_country_stock_record = {
			country = c:FRA
			market = scope:modeu5_capacity_test_market
			good = wheat
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_test_wrapped_wheat_total
			value = scope:modeu5_read_country_stock_cap
		}

		modeu5_read_country_stock_record = {
			country = c:FRA
			market = scope:modeu5_capacity_test_market
			good = iron
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_test_wrapped_iron_total
			value = scope:modeu5_read_country_stock_cap
		}

		modeu5_recalculate_country_market_capacity = {
			country = c:FRA
			market = scope:modeu5_capacity_test_market
			good = wheat
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_test_wheat_total
			value = scope:modeu5_capacity_total
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_test_wheat_base
			value = scope:modeu5_capacity_base
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_test_wheat_trade
			value = scope:modeu5_capacity_trade_capacity
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_test_wheat_location_rank
			value = scope:modeu5_capacity_location_rank_capacity
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_test_wheat_location_count
			value = scope:modeu5_capacity_location_count
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_test_wheat_building
			value = scope:modeu5_capacity_building
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_test_wheat_foreign
			value = scope:modeu5_capacity_foreign
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_test_expected_available
			value = {
				value = scope:modeu5_capacity_total
				subtract = scope:modeu5_capacity_test_wheat_stock_before
				min = 0
			}
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_test_expected_over_cap
			value = {
				value = scope:modeu5_capacity_test_wheat_stock_before
				subtract = scope:modeu5_capacity_total
				min = 0
			}
		}

		modeu5_read_country_stock_record = {
			country = c:FRA
			market = scope:modeu5_capacity_test_market
			good = wheat
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_test_wheat_stock_after
			value = scope:modeu5_read_country_stock
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_test_wheat_read_total
			value = scope:modeu5_read_country_stock_cap
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_test_wheat_read_base
			value = scope:modeu5_read_country_base_capacity
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_test_wheat_read_building
			value = scope:modeu5_read_country_building_capacity
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_test_wheat_read_foreign
			value = scope:modeu5_read_country_foreign_capacity
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_test_wheat_read_available
			value = scope:modeu5_read_available_capacity
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_test_wheat_read_over_cap
			value = scope:modeu5_read_over_capacity
		}

		save_temporary_scope_value_as = {
			name = modeu5_capacity_test_reconciled_total
			value = {
				value = scope:modeu5_capacity_test_wheat_base
				add = scope:modeu5_capacity_test_wheat_building
				add = scope:modeu5_capacity_test_wheat_foreign
			}
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_test_reconciled_base
			value = {
				value = scope:modeu5_capacity_test_wheat_trade
				add = scope:modeu5_capacity_test_wheat_location_rank
			}
		}
		set_global_variable = {
			name = modeu5_debug_us02_dump_fra_wheat_stock
			value = scope:modeu5_capacity_test_wheat_stock_after
		}
		set_global_variable = {
			name = modeu5_debug_us02_dump_fra_wheat_capacity
			value = scope:modeu5_capacity_test_wheat_total
		}
		set_global_variable = {
			name = modeu5_debug_us02_dump_fra_wheat_available
			value = scope:modeu5_capacity_test_wheat_read_available
		}
		set_global_variable = {
			name = modeu5_debug_us02_dump_fra_wheat_over_cap
			value = scope:modeu5_capacity_test_wheat_read_over_cap
		}
		set_global_variable = {
			name = modeu5_debug_us02_dump_fra_wheat_trade_capacity
			value = scope:modeu5_capacity_test_wheat_trade
		}
		set_global_variable = {
			name = modeu5_debug_us02_dump_fra_wheat_location_rank_capacity
			value = scope:modeu5_capacity_test_wheat_location_rank
		}
		set_global_variable = {
			name = modeu5_debug_us02_dump_fra_wheat_location_count
			value = scope:modeu5_capacity_test_wheat_location_count
		}
		set_global_variable = {
			name = modeu5_debug_us02_dump_fra_iron_capacity
			value = scope:modeu5_capacity_test_wrapped_iron_total
		}
		modeu5_debug_write_us02_capacity_dump_variables = yes

		if = {
			limit = {
				scope:modeu5_capacity_test_wheat_total > 0
				scope:modeu5_capacity_test_wheat_location_rank > 0
				scope:modeu5_capacity_test_wheat_location_count > 0
				scope:modeu5_capacity_test_wheat_total = scope:modeu5_capacity_test_reconciled_total
				scope:modeu5_capacity_test_wheat_base = scope:modeu5_capacity_test_reconciled_base
				scope:modeu5_capacity_test_wrapped_wheat_total = scope:modeu5_capacity_test_wheat_total
				scope:modeu5_capacity_test_wrapped_iron_total = scope:modeu5_capacity_test_wheat_total
				scope:modeu5_capacity_test_wheat_read_total = scope:modeu5_capacity_test_wheat_total
				scope:modeu5_capacity_test_wheat_read_base = scope:modeu5_capacity_test_wheat_base
				scope:modeu5_capacity_test_wheat_read_building = scope:modeu5_capacity_test_wheat_building
				scope:modeu5_capacity_test_wheat_read_foreign = scope:modeu5_capacity_test_wheat_foreign
				scope:modeu5_capacity_test_wheat_stock_after = scope:modeu5_capacity_test_wheat_stock_before
				scope:modeu5_capacity_test_wheat_read_available = scope:modeu5_capacity_test_expected_available
				scope:modeu5_capacity_test_wheat_read_over_cap = scope:modeu5_capacity_test_expected_over_cap
			}
			set_global_variable = modeu5_test_storage_capacity_passed
		}
		else = {
			remove_global_variable = modeu5_test_storage_capacity_passed
			error_log = "ModeU5 deterministic US-02 storage-capacity test failed."
		}
	}
	else = {
		remove_global_variable = modeu5_test_storage_capacity_passed
		error_log = "ModeU5 deterministic US-02 storage-capacity test requires FRA."
	}
}

modeu5_debug_run_storage_capacity_timing_probe = {
	save_temporary_scope_as = modeu5_capacity_test_controller

	remove_global_variable = modeu5_test_storage_capacity_timing_immediate_passed
	remove_global_variable = modeu5_test_storage_capacity_timing_monthly_passed
	remove_global_variable = modeu5_test_storage_capacity_timing_pending
	remove_global_variable = modeu5_test_storage_capacity_timing_failed

	if = {
		limit = { country_exists = c:FRA }
		c:FRA.capital.market = { save_temporary_scope_as = modeu5_capacity_probe_market }

		modeu5_recalculate_country_market_capacity = {
			country = c:FRA
			market = scope:modeu5_capacity_probe_market
			good = wheat
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_probe_before_capacity
			value = scope:modeu5_capacity_total
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_probe_before_trade
			value = scope:modeu5_capacity_trade_capacity
		}

		c:FRA.capital = {
			change_building_level_in_location = {
				building = building_type:marketplace
				value = 10
			}
		}

		modeu5_recalculate_country_market_capacity = {
			country = c:FRA
			market = scope:modeu5_capacity_probe_market
			good = wheat
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_probe_after_build_capacity
			value = scope:modeu5_capacity_total
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_probe_after_build_trade
			value = scope:modeu5_capacity_trade_capacity
		}
		save_temporary_scope_value_as = { name = modeu5_capacity_probe_after_month_capacity value = 0 }
		save_temporary_scope_value_as = { name = modeu5_capacity_probe_after_month_trade value = 0 }
		save_temporary_scope_value_as = {
			name = modeu5_capacity_probe_delta_immediate_capacity
			value = {
				value = scope:modeu5_capacity_probe_after_build_capacity
				subtract = scope:modeu5_capacity_probe_before_capacity
			}
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_probe_delta_immediate_trade
			value = {
				value = scope:modeu5_capacity_probe_after_build_trade
				subtract = scope:modeu5_capacity_probe_before_trade
			}
		}
		save_temporary_scope_value_as = { name = modeu5_capacity_probe_delta_month_capacity value = 0 }
		save_temporary_scope_value_as = { name = modeu5_capacity_probe_delta_month_trade value = 0 }

		set_global_variable = modeu5_debug_us02_probe_started
		set_global_variable = { name = modeu5_debug_us02_probe_before_capacity value = scope:modeu5_capacity_probe_before_capacity }
		set_global_variable = { name = modeu5_debug_us02_probe_before_trade value = scope:modeu5_capacity_probe_before_trade }
		set_global_variable = { name = modeu5_debug_us02_probe_after_build_capacity value = scope:modeu5_capacity_probe_after_build_capacity }
		set_global_variable = { name = modeu5_debug_us02_probe_after_build_trade value = scope:modeu5_capacity_probe_after_build_trade }
		set_global_variable = { name = modeu5_debug_us02_probe_after_month_capacity value = 0 }
		set_global_variable = { name = modeu5_debug_us02_probe_after_month_trade value = 0 }
		set_global_variable = { name = modeu5_debug_us02_probe_delta_immediate_capacity value = scope:modeu5_capacity_probe_delta_immediate_capacity }
		set_global_variable = { name = modeu5_debug_us02_probe_delta_immediate_trade value = scope:modeu5_capacity_probe_delta_immediate_trade }
		set_global_variable = { name = modeu5_debug_us02_probe_delta_month_capacity value = 0 }
		set_global_variable = { name = modeu5_debug_us02_probe_delta_month_trade value = 0 }
		modeu5_debug_write_us02_timing_probe_variables = yes

		if = {
			limit = {
				scope:modeu5_capacity_probe_delta_immediate_capacity > 0
				scope:modeu5_capacity_probe_delta_immediate_trade > 0
			}
			set_global_variable = modeu5_test_storage_capacity_timing_immediate_passed
		}
		else = {
			set_global_variable = modeu5_test_storage_capacity_timing_pending
		}
	}
	else = {
		remove_global_variable = modeu5_debug_us02_probe_started
		save_temporary_scope_value_as = { name = modeu5_capacity_probe_before_capacity value = 0 }
		save_temporary_scope_value_as = { name = modeu5_capacity_probe_after_build_capacity value = 0 }
		save_temporary_scope_value_as = { name = modeu5_capacity_probe_after_month_capacity value = 0 }
		save_temporary_scope_value_as = { name = modeu5_capacity_probe_delta_immediate_capacity value = 0 }
		save_temporary_scope_value_as = { name = modeu5_capacity_probe_delta_month_capacity value = 0 }
		save_temporary_scope_value_as = { name = modeu5_capacity_probe_before_trade value = 0 }
		save_temporary_scope_value_as = { name = modeu5_capacity_probe_after_build_trade value = 0 }
		save_temporary_scope_value_as = { name = modeu5_capacity_probe_after_month_trade value = 0 }
		save_temporary_scope_value_as = { name = modeu5_capacity_probe_delta_immediate_trade value = 0 }
		save_temporary_scope_value_as = { name = modeu5_capacity_probe_delta_month_trade value = 0 }
		modeu5_debug_write_us02_timing_probe_variables = yes
		set_global_variable = modeu5_test_storage_capacity_timing_failed
		error_log = "ModeU5 deterministic US-02 storage-capacity timing probe requires FRA."
	}
}

modeu5_debug_finish_storage_capacity_timing_probe = {
	save_temporary_scope_as = modeu5_capacity_test_controller

	remove_global_variable = modeu5_test_storage_capacity_timing_monthly_passed
	remove_global_variable = modeu5_test_storage_capacity_timing_failed

	if = {
		limit = {
			country_exists = c:FRA
			has_global_variable = modeu5_debug_us02_probe_started
			has_global_variable = modeu5_debug_us02_probe_before_capacity
			has_global_variable = modeu5_debug_us02_probe_before_trade
		}
		c:FRA.capital.market = { save_temporary_scope_as = modeu5_capacity_probe_market }

		modeu5_read_country_stock_record = {
			country = c:FRA
			market = scope:modeu5_capacity_probe_market
			good = wheat
		}
		save_temporary_scope_value_as = { name = modeu5_capacity_probe_before_capacity value = global_var:modeu5_debug_us02_probe_before_capacity }
		save_temporary_scope_value_as = { name = modeu5_capacity_probe_before_trade value = global_var:modeu5_debug_us02_probe_before_trade }
		save_temporary_scope_value_as = { name = modeu5_capacity_probe_after_build_capacity value = global_var:modeu5_debug_us02_probe_after_build_capacity }
		save_temporary_scope_value_as = { name = modeu5_capacity_probe_after_build_trade value = global_var:modeu5_debug_us02_probe_after_build_trade }
		save_temporary_scope_value_as = { name = modeu5_capacity_probe_after_month_capacity value = scope:modeu5_read_country_stock_cap }
		save_temporary_scope_value_as = {
			name = modeu5_capacity_probe_after_month_trade
			value = {
				value = "scope:modeu5_capacity_probe_market.merchant_capacity(scope:modeu5_country)"
				multiply = modeu5_storage_capacity_per_trade_capacity
				min = 0
			}
		}
		save_temporary_scope_value_as = { name = modeu5_capacity_probe_delta_immediate_capacity value = global_var:modeu5_debug_us02_probe_delta_immediate_capacity }
		save_temporary_scope_value_as = { name = modeu5_capacity_probe_delta_immediate_trade value = global_var:modeu5_debug_us02_probe_delta_immediate_trade }
		save_temporary_scope_value_as = {
			name = modeu5_capacity_probe_delta_month_capacity
			value = {
				value = scope:modeu5_capacity_probe_after_month_capacity
				subtract = scope:modeu5_capacity_probe_before_capacity
			}
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_probe_delta_month_trade
			value = {
				value = scope:modeu5_capacity_probe_after_month_trade
				subtract = scope:modeu5_capacity_probe_before_trade
			}
		}

		set_global_variable = { name = modeu5_debug_us02_probe_after_month_capacity value = scope:modeu5_capacity_probe_after_month_capacity }
		set_global_variable = { name = modeu5_debug_us02_probe_after_month_trade value = scope:modeu5_capacity_probe_after_month_trade }
		set_global_variable = { name = modeu5_debug_us02_probe_delta_month_capacity value = scope:modeu5_capacity_probe_delta_month_capacity }
		set_global_variable = { name = modeu5_debug_us02_probe_delta_month_trade value = scope:modeu5_capacity_probe_delta_month_trade }
		modeu5_debug_write_us02_timing_probe_variables = yes

		if = {
			limit = {
				scope:modeu5_capacity_probe_delta_month_capacity > 0
				scope:modeu5_capacity_probe_delta_month_trade > 0
			}
			remove_global_variable = modeu5_test_storage_capacity_timing_pending
			set_global_variable = modeu5_test_storage_capacity_timing_monthly_passed
		}
		else = {
			remove_global_variable = modeu5_test_storage_capacity_timing_pending
			set_global_variable = modeu5_test_storage_capacity_timing_failed
			error_log = "ModeU5 deterministic US-02 storage-capacity timing probe failed after monthly tick."
		}
	}
	else = {
		save_temporary_scope_value_as = { name = modeu5_capacity_probe_before_capacity value = 0 }
		save_temporary_scope_value_as = { name = modeu5_capacity_probe_after_build_capacity value = 0 }
		save_temporary_scope_value_as = { name = modeu5_capacity_probe_after_month_capacity value = 0 }
		save_temporary_scope_value_as = { name = modeu5_capacity_probe_delta_immediate_capacity value = 0 }
		save_temporary_scope_value_as = { name = modeu5_capacity_probe_delta_month_capacity value = 0 }
		save_temporary_scope_value_as = { name = modeu5_capacity_probe_before_trade value = 0 }
		save_temporary_scope_value_as = { name = modeu5_capacity_probe_after_build_trade value = 0 }
		save_temporary_scope_value_as = { name = modeu5_capacity_probe_after_month_trade value = 0 }
		save_temporary_scope_value_as = { name = modeu5_capacity_probe_delta_immediate_trade value = 0 }
		save_temporary_scope_value_as = { name = modeu5_capacity_probe_delta_month_trade value = 0 }
		modeu5_debug_write_us02_timing_probe_variables = yes
		set_global_variable = modeu5_test_storage_capacity_timing_failed
		error_log = "ModeU5 deterministic US-02 storage-capacity timing probe was finished without a started probe."
	}
}
