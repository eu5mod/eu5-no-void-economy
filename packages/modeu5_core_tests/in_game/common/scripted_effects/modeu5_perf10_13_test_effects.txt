modeu5_debug_run_perf10_13_active_repair_metrics_probe = {
	modeu5_perf10_13_clear_test_results = yes
	save_temporary_scope_as = modeu5_perf10_13_controller
	save_temporary_scope_as = modeu5_active_repair_controller

	if = {
		limit = { country_exists = c:FRA }
		c:FRA.capital.market = { save_temporary_scope_as = modeu5_perf10_13_test_market }

		modeu5_add_stock = {
			country = c:FRA
			market = scope:modeu5_perf10_13_test_market
			good = wheat
			quantity = 5
			capacity_policy = allow_over_capacity
		}

		modeu5_clear_all_active_stock_markets = yes
		modeu5_perf13_reset_active_repair_metrics = yes

		modeu5_perf10_13_log_active_repair_dump = yes

		if = {
			limit = {
				has_global_variable_list = modeu5_active_markets_any_good
				has_global_variable_list = modeu5_wheat_active_markets
				is_target_in_global_variable_list = {
					name = modeu5_active_markets_any_good
					target = scope:modeu5_perf10_13_test_market
				}
				is_target_in_global_variable_list = {
					name = modeu5_wheat_active_markets
					target = scope:modeu5_perf10_13_test_market

				global_var:modeu5_perf13_active_repair_market_good_scans >= 1

			debug_log = "ModeU5 PERF-11 RESULT active_repair PASS"
			debug_log = "ModeU5 TEST PASS scenario=perf10_13_active_repair_metrics"
		}

			set_global_variable = modeu5_test_perf10_13_active_repair_failed
			debug_log = "ModeU5 PERF-11 RESULT active_repair FAIL"
			debug_log = "ModeU5 TEST FAIL scenario=perf10_13_active_repair_metrics"
			error_log = "ModeU5 PERF-11 active-list repair probe failed."
		}

		modeu5_remove_stock = {
			country = c:FRA
			market = scope:modeu5_perf10_13_test_market
			good = wheat
			quantity = 5
			reason = debug_test
		}
		modeu5_clear_all_active_stock_markets = yes
		modeu5_clear_all_dirty_stock_markets = yes

	else = {
		modeu5_perf13_reset_active_repair_metrics = yes
		modeu5_perf10_13_write_active_repair_dump_variables = yes
		modeu5_perf10_13_log_active_repair_dump = yes

		debug_log = "ModeU5 TEST BLOCKED scenario=perf10_13_active_repair_metrics reason=missing_FRA"
	}
