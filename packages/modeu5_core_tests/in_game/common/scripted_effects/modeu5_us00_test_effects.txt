	set_global_variable = modeu5_test_us00_controlled_e2e_started
	save_temporary_scope_as = modeu5_us00_test_controller

	if = {
		limit = { country_exists = c:FRA }
		c:FRA.capital.market = { save_temporary_scope_as = modeu5_us00_test_market }

		modeu5_clear_us00_record = {
			country = c:FRA
			market = scope:modeu5_us00_test_market
			good = wheat
		}
		modeu5_debug_replace_country_wheat_capacity = {
			country = c:FRA
			market = scope:modeu5_us00_test_market
			value = 40
		}
		modeu5_remove_stock = {
			country = c:FRA
			market = scope:modeu5_us00_test_market
			good = wheat
			quantity = 1000000
			reason = debug_test
		}
		modeu5_add_stock = {
			country = c:FRA
			market = scope:modeu5_us00_test_market
			good = wheat
			quantity = 100
			capacity_policy = enforce
		}
		save_temporary_scope_value_as = {
			name = modeu5_us00_test_actual_added
			value = scope:modeu5_actual_added_quantity
		}
		save_temporary_scope_value_as = {
			name = modeu5_us00_test_actual_rejected
			value = scope:modeu5_rejected_quantity
		}

		modeu5_update_production_rejection_ledger = {
			country = c:FRA
			market = scope:modeu5_us00_test_market
			good = wheat
			produced_quantity = 100
			actual_added_quantity = scope:modeu5_us00_test_actual_added
			rejected_quantity = scope:modeu5_us00_test_actual_rejected
		}
		modeu5_run_us00_record_calculations = {
			country = c:FRA
			market = scope:modeu5_us00_test_market
			good = wheat
			good_price = 2
			good_price_source = 99
		}

		modeu5_read_country_stock_record = {
			country = c:FRA
			market = scope:modeu5_us00_test_market
			good = wheat
		}
		save_temporary_scope_value_as = { name = modeu5_us00_test_stock_after value = scope:modeu5_read_country_stock }
		save_temporary_scope_value_as = { name = modeu5_us00_test_capacity value = scope:modeu5_read_country_stock_cap }
		save_temporary_scope_value_as = { name = modeu5_us00_test_market_stock_after value = scope:modeu5_read_market_stock }

		modeu5_read_void_economy_record = {
			country = c:FRA
			market = scope:modeu5_us00_test_market
			good = wheat
		}
		save_temporary_scope_value_as = { name = modeu5_us00_test_produced value = scope:modeu5_us00_produced }
		save_temporary_scope_value_as = { name = modeu5_us00_test_added value = scope:modeu5_us00_added }
		save_temporary_scope_value_as = { name = modeu5_us00_test_rejected value = scope:modeu5_us00_rejected }
		save_temporary_scope_value_as = { name = modeu5_us00_test_ratio value = scope:modeu5_us00_read_overproduction_ratio }
		save_temporary_scope_value_as = { name = modeu5_us00_test_effective_ratio value = scope:modeu5_us00_read_effective_overproduction_ratio }
		save_temporary_scope_value_as = { name = modeu5_us00_test_void_wealth value = scope:modeu5_us00_read_void_wealth }
		save_temporary_scope_value_as = { name = modeu5_us00_test_taxable_proxy value = scope:modeu5_us00_read_void_taxable_income_proxy }
		save_temporary_scope_value_as = { name = modeu5_us00_test_penalty value = scope:modeu5_us00_read_production_penalty }
		c:FRA = {
			if = {
				limit = {
					has_variable_map = modeu5_void_wealth_by_market
					is_key_in_variable_map = {
						name = modeu5_void_wealth_by_market
						target = scope:modeu5_us00_test_market
					}
				}
				save_temporary_scope_value_as = {
					name = modeu5_us00_test_market_void_wealth
					value = "variable_map(modeu5_void_wealth_by_market|scope:modeu5_us00_test_market)"
				}
			}
			else = {
				save_temporary_scope_value_as = { name = modeu5_us00_test_market_void_wealth value = 0 }
			}
			if = {
				limit = { has_variable = modeu5_total_void_wealth }
				save_temporary_scope_value_as = { name = modeu5_us00_test_total_void_wealth value = var:modeu5_total_void_wealth }
			}
			else = {
				save_temporary_scope_value_as = { name = modeu5_us00_test_total_void_wealth value = 0 }
			}
		}
		save_temporary_scope_value_as = {
			name = modeu5_us00_test_expected_ratio
			value = {
				value = 60
				divide = 100
			}
		}
		save_temporary_scope_value_as = {
			name = modeu5_us00_test_expected_effective_ratio
			value = {
				value = scope:modeu5_us00_test_expected_ratio
				subtract = modeu5_overproduction_stability_buffer
				min = 0
			}
		}
		save_temporary_scope_value_as = {
			name = modeu5_us00_test_expected_penalty_magnitude
			value = {
				value = scope:modeu5_us00_test_expected_effective_ratio
				multiply = modeu5_production_penalty_per_effective_overproduction
				max = modeu5_max_production_penalty
			}
		}
		save_temporary_scope_value_as = {
			name = modeu5_us00_test_penalty_delta
			value = {
				value = scope:modeu5_us00_test_penalty
				add = scope:modeu5_us00_test_expected_penalty_magnitude
			}
		}

		modeu5_us00_write_e2e_dump_variables = yes
		modeu5_us00_log_e2e_dump = yes

		if = {
			limit = {
				scope:modeu5_us00_test_actual_added = 40
				scope:modeu5_us00_test_actual_rejected = 60
				scope:modeu5_us00_test_stock_after = 40
				scope:modeu5_us00_test_capacity = 40
				scope:modeu5_us00_test_produced = 100
				scope:modeu5_us00_test_added = 40
				scope:modeu5_us00_test_rejected = 60
				scope:modeu5_us00_test_ratio = scope:modeu5_us00_test_expected_ratio
				scope:modeu5_us00_test_effective_ratio = scope:modeu5_us00_test_expected_effective_ratio
				scope:modeu5_us00_test_void_wealth = 120
				scope:modeu5_us00_test_taxable_proxy = 120
				scope:modeu5_us00_test_penalty_delta = 0
				scope:modeu5_us00_test_market_void_wealth = 120
				scope:modeu5_us00_test_total_void_wealth = 120
			}
			set_global_variable = modeu5_test_us00_controlled_e2e_passed
			debug_log = "ModeU5 US-00 RESULT controlled_e2e PASS"
		}
		else = {
			set_global_variable = modeu5_test_us00_controlled_e2e_failed
			debug_log = "ModeU5 US-00 RESULT controlled_e2e FAIL"
			error_log = "ModeU5 deterministic US-00 controlled pipeline test failed."
		}

		modeu5_remove_stock = {
			country = c:FRA
			market = scope:modeu5_us00_test_market
			good = wheat
			quantity = 1000000
			reason = debug_test
		}
		modeu5_clear_us00_record = {
			country = c:FRA
			market = scope:modeu5_us00_test_market
			good = wheat
		}
	}
	else = {
		set_global_variable = modeu5_test_us00_controlled_e2e_blocked
		debug_log = "ModeU5 US-00 RESULT controlled_e2e BLOCKED missing_FRA"
		error_log = "ModeU5 deterministic US-00 controlled pipeline test requires FRA."
	}
}

modeu5_us00_debug_run_probe021 = {
	modeu5_us00_clear_test_results = yes
	set_global_variable = modeu5_test_probe021_started
	save_temporary_scope_as = modeu5_us00_test_controller

	if = {
		limit = { country_exists = c:FRA }
		c:FRA = {
			every_owned_location = {
				modeu5_probe021_increment_location_count = yes
				save_temporary_scope_value_as = {
					name = modeu5_probe021_location_wheat_goods_output
					value = {
						value = "goods_output(goods:wheat)"
						min = 0
					}
				}
				save_temporary_scope_value_as = {
					name = modeu5_probe021_location_raw_material_output
					value = {
						value = raw_material_output
						min = 0
					}
				}
				if = {
					limit = { scope:modeu5_probe021_location_wheat_goods_output > 0 }
					modeu5_probe021_increment_positive_goods_output_count = yes
				}
				modeu5_probe021_add_wheat_goods_output_total = yes
				modeu5_probe021_add_raw_material_output_total = yes
			}
		}

		modeu5_us00_write_probe021_dump_variables = yes
		modeu5_us00_log_probe021_dump = yes
		if = {
			limit = {
				NOT = { has_global_variable = modeu5_us00_probe021_location_count }
			}
			set_global_variable = modeu5_test_probe021_failed
			debug_log = "ModeU5 PROBE-021 RESULT location_output FAIL missing_location_counter"
			error_log = "ModeU5 PROBE-021 did not produce a location counter after iterating FRA owned locations."
		}
		else_if = {
			limit = {
				NOT = { global_var:modeu5_us00_probe021_location_count > 0 }
			}
			set_global_variable = modeu5_test_probe021_failed
			debug_log = "ModeU5 PROBE-021 RESULT location_output FAIL no_owned_locations_counted"
			error_log = "ModeU5 PROBE-021 counted zero FRA owned locations."
		}
		else_if = {
			limit = {
				has_global_variable = modeu5_us00_probe021_positive_goods_output_count
				global_var:modeu5_us00_probe021_location_count > 0
				global_var:modeu5_us00_probe021_positive_goods_output_count > 0
			}
			set_global_variable = modeu5_test_probe021_passed
			debug_log = "ModeU5 PROBE-021 RESULT location_output PASS"
		}
		else = {
			set_global_variable = modeu5_test_probe021_pending
			debug_log = "ModeU5 PROBE-021 RESULT location_output PENDING no_positive_wheat_output"
		}
	}
	else = {
		set_global_variable = modeu5_test_probe021_failed
		debug_log = "ModeU5 PROBE-021 RESULT location_output FAIL missing_FRA"
		error_log = "ModeU5 PROBE-021 requires FRA."
	}
}
