modeu5_perf03_probe_current_market_country_cache = {
	modeu5_perf03_clear_probe_results = yes
	set_global_variable = modeu5_test_perf03_market_country_cache_started
	save_temporary_scope_as = modeu5_perf03_probe_country

	if = {
		limit = { has_markets = no }
		set_global_variable = modeu5_test_perf03_market_country_cache_blocked
		error_log = "ModeU5 PERF-03 market-country cache probe blocked because the current country has no markets."
	}
	else = {
		capital.market = { save_temporary_scope_as = modeu5_perf03_probe_market }

		scope:modeu5_perf03_probe_market = {
			save_temporary_scope_as = modeu5_market_country_cache_dirty_market
		}
		modeu5_mark_market_country_cache_dirty = yes
		modeu5_repair_dirty_market_country_caches = yes

		scope:modeu5_perf03_probe_market = {
			save_temporary_scope_as = modeu5_market_country_cache_market
		}
		modeu5_rebuild_countries_present_in_market = yes

		if = {
			limit = {
				OR = {
					NOT = { has_global_variable_list = modeu5_countries_present_in_market }
					NOT = {
						is_target_in_global_variable_list = {
							name = modeu5_countries_present_in_market
							target = scope:modeu5_perf03_probe_country
						}
					}
				}
			}
			set_global_variable = modeu5_test_perf03_market_country_cache_missing_current_country
			set_global_variable = modeu5_test_perf03_market_country_cache_failed
			error_log = "ModeU5 PERF-03 market-country cache probe did not find the current country in its capital market cache."
		}
		else = {
			set_global_variable = modeu5_test_perf03_market_country_cache_passed
		}
	}

	modeu5_perf03_mirror_probe_to_ui = yes
	modeu5_perf03_log_probe_result = yes
}

modeu5_perf03_probe_nonterritorial_market_presence = {
	modeu5_perf03_clear_nonterritorial_presence_probe_results = yes
	set_global_variable = modeu5_test_perf03_nonterritorial_presence_started
	save_temporary_scope_as = modeu5_perf03_presence_country

	set_local_variable = { name = modeu5_perf03_presence_native_markets_accumulator value = 0 }
	set_local_variable = { name = modeu5_perf03_presence_territorial_markets_accumulator value = 0 }
	set_local_variable = { name = modeu5_perf03_presence_nonterritorial_markets_accumulator value = 0 }
	set_local_variable = { name = modeu5_perf03_presence_nonterritorial_with_capacity_accumulator value = 0 }
	set_local_variable = { name = modeu5_perf03_presence_nonterritorial_with_stock_accumulator value = 0 }
	set_local_variable = { name = modeu5_perf03_presence_total_capacity_accumulator value = 0 }
	set_local_variable = { name = modeu5_perf03_presence_total_stock_accumulator value = 0 }
	set_local_variable = { name = modeu5_perf03_presence_total_trade_capacity_accumulator value = 0 }
	set_local_variable = { name = modeu5_perf03_presence_total_foreign_capacity_accumulator value = 0 }
	set_local_variable = { name = modeu5_perf03_presence_cache_hits_accumulator value = 0 }
	set_local_variable = { name = modeu5_perf03_presence_cache_misses_accumulator value = 0 }
	set_local_variable = { name = modeu5_perf03_presence_current_market_owned_locations value = 0 }

	if = {
		limit = { has_markets = no }
		set_global_variable = modeu5_test_perf03_nonterritorial_presence_blocked
		set_global_variable = modeu5_test_perf03_nonterritorial_presence_no_markets
		error_log = "ModeU5 PERF-03 non-territorial market presence probe blocked because the current country has no markets."
	}
	else = {
		every_market_present_in_country = {
			save_temporary_scope_as = modeu5_perf03_presence_market
			scope:modeu5_perf03_presence_country = {
				change_local_variable = { name = modeu5_perf03_presence_native_markets_accumulator add = 1 }
				set_local_variable = { name = modeu5_perf03_presence_current_market_owned_locations value = 0 }
			}

			scope:modeu5_perf03_presence_country = {
				every_owned_location = {
					limit = { market = scope:modeu5_perf03_presence_market }
					scope:modeu5_perf03_presence_country = {
						change_local_variable = {
							name = modeu5_perf03_presence_current_market_owned_locations
							add = 1
						}
					}
				}

				save_temporary_scope_value_as = {
					name = modeu5_perf03_presence_current_market_owned_locations_value
					value = local_var:modeu5_perf03_presence_current_market_owned_locations
				}
			}

			if = {
				limit = { scope:modeu5_perf03_presence_current_market_owned_locations_value > 0 }
				scope:modeu5_perf03_presence_country = {
					change_local_variable = { name = modeu5_perf03_presence_territorial_markets_accumulator add = 1 }
				}
			}
			else = {
				scope:modeu5_perf03_presence_country = {
					change_local_variable = {
						name = modeu5_perf03_presence_nonterritorial_markets_accumulator
						add = 1
					}
				}

				modeu5_recalculate_country_market_capacity = {
					country = scope:modeu5_perf03_presence_country
					market = scope:modeu5_perf03_presence_market
					good = wheat
				}
				modeu5_read_country_stock_record = {
					country = scope:modeu5_perf03_presence_country
					market = scope:modeu5_perf03_presence_market
					good = wheat
				}

				scope:modeu5_perf03_presence_country = {
					change_local_variable = {
						name = modeu5_perf03_presence_total_capacity_accumulator
						add = scope:modeu5_capacity_total
					}
					change_local_variable = {
						name = modeu5_perf03_presence_total_trade_capacity_accumulator
						add = scope:modeu5_capacity_trade_capacity
					}
					change_local_variable = {
						name = modeu5_perf03_presence_total_stock_accumulator
						add = scope:modeu5_read_country_stock
					}
					change_local_variable = {
						name = modeu5_perf03_presence_total_foreign_capacity_accumulator
						add = scope:modeu5_capacity_foreign
					}
				}

				if = {
					limit = { scope:modeu5_capacity_total > 0 }
					scope:modeu5_perf03_presence_country = {
						change_local_variable = {
							name = modeu5_perf03_presence_nonterritorial_with_capacity_accumulator
							add = 1
						}
					}
				}
				if = {
					limit = { scope:modeu5_read_country_stock > 0 }
					scope:modeu5_perf03_presence_country = {
						change_local_variable = {
							name = modeu5_perf03_presence_nonterritorial_with_stock_accumulator
							add = 1
						}
					}
				}

				scope:modeu5_perf03_presence_market = {
					save_temporary_scope_as = modeu5_market_country_cache_market
				}
				modeu5_rebuild_countries_present_in_market = yes

				if = {
					limit = {
						has_global_variable_list = modeu5_countries_present_in_market
						is_target_in_global_variable_list = {
							name = modeu5_countries_present_in_market
							target = scope:modeu5_perf03_presence_country
						}
					}
					scope:modeu5_perf03_presence_country = {
						change_local_variable = {
							name = modeu5_perf03_presence_cache_hits_accumulator
							add = 1
						}
					}
				}
				else = {
					scope:modeu5_perf03_presence_country = {
						change_local_variable = {
							name = modeu5_perf03_presence_cache_misses_accumulator
							add = 1
						}
					}

					if = {
						limit = {
							OR = {
								scope:modeu5_capacity_total > 0
								scope:modeu5_read_country_stock > 0
							}
						}
						set_global_variable = modeu5_test_perf03_nonterritorial_presence_stock_capable_missed
					}
				}
			}
		}
	}

	save_temporary_scope_value_as = {
		name = modeu5_perf03_presence_native_markets
		value = local_var:modeu5_perf03_presence_native_markets_accumulator
	}
	save_temporary_scope_value_as = {
		name = modeu5_perf03_presence_territorial_markets
		value = local_var:modeu5_perf03_presence_territorial_markets_accumulator
	}
	save_temporary_scope_value_as = {
		name = modeu5_perf03_presence_nonterritorial_markets
		value = local_var:modeu5_perf03_presence_nonterritorial_markets_accumulator
	}
	save_temporary_scope_value_as = {
		name = modeu5_perf03_presence_nonterritorial_with_capacity
		value = local_var:modeu5_perf03_presence_nonterritorial_with_capacity_accumulator
	}
	save_temporary_scope_value_as = {
		name = modeu5_perf03_presence_nonterritorial_with_stock
		value = local_var:modeu5_perf03_presence_nonterritorial_with_stock_accumulator
	}
	save_temporary_scope_value_as = {
		name = modeu5_perf03_presence_total_capacity
		value = local_var:modeu5_perf03_presence_total_capacity_accumulator
	}
	save_temporary_scope_value_as = {
		name = modeu5_perf03_presence_total_stock
		value = local_var:modeu5_perf03_presence_total_stock_accumulator
	}
	save_temporary_scope_value_as = {
		name = modeu5_perf03_presence_total_trade_capacity
		value = local_var:modeu5_perf03_presence_total_trade_capacity_accumulator
	}
	save_temporary_scope_value_as = {
		name = modeu5_perf03_presence_total_foreign_capacity
		value = local_var:modeu5_perf03_presence_total_foreign_capacity_accumulator
	}
	save_temporary_scope_value_as = {
		name = modeu5_perf03_presence_cache_hits
		value = local_var:modeu5_perf03_presence_cache_hits_accumulator
	}
	save_temporary_scope_value_as = {
		name = modeu5_perf03_presence_cache_misses
		value = local_var:modeu5_perf03_presence_cache_misses_accumulator
	}

	if = {
		limit = {
			NOT = { has_global_variable = modeu5_test_perf03_nonterritorial_presence_blocked }
			scope:modeu5_perf03_presence_nonterritorial_markets = 0
		}
		set_global_variable = modeu5_test_perf03_nonterritorial_presence_blocked
		set_global_variable = modeu5_test_perf03_nonterritorial_presence_no_candidate
	}
	else_if = {
		limit = { has_global_variable = modeu5_test_perf03_nonterritorial_presence_stock_capable_missed }
		set_global_variable = modeu5_test_perf03_nonterritorial_presence_risk
		error_log = "ModeU5 PERF-03 non-territorial market presence probe found stock-capable country-market presence missed by the location-owner cache."
	}
	else_if = {
		limit = { NOT = { has_global_variable = modeu5_test_perf03_nonterritorial_presence_blocked } }
		set_global_variable = modeu5_test_perf03_nonterritorial_presence_passed
	}

	modeu5_perf03_mirror_nonterritorial_presence_probe_to_ui = yes
	modeu5_perf03_log_nonterritorial_presence_probe_result = yes
}
