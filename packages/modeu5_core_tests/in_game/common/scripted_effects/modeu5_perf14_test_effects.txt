# PERF-14 probes CMM main-mode plumbing and the first accounting decision gate.

modeu5_perf14_clear_test_results = {
	remove_global_variable = modeu5_test_perf14_performance_mode_cmm_started
	remove_global_variable = modeu5_test_perf14_performance_mode_cmm_passed
	remove_global_variable = modeu5_test_perf14_performance_mode_cmm_failed
	remove_global_variable = modeu5_test_perf14_performance_mode_cmm_blocked

	remove_variable = modeu5_perf14_ui_main_mode
	remove_variable = modeu5_perf14_ui_performance_enabled
	remove_variable = modeu5_perf14_ui_normal_enabled
	remove_variable = modeu5_perf14_ui_deactivated
	remove_variable = modeu5_perf14_ui_detailed_enabled
	remove_variable = modeu5_perf14_ui_fallback_required
	remove_variable = modeu5_perf14_ui_country_is_human
	remove_variable = modeu5_perf14_ui_market_human_relevant
	remove_variable = modeu5_perf14_ui_human_relevant_market_count
	remove_variable = modeu5_perf14_ui_current_market_human_relevant
}

modeu5_perf14_set_cmm_main_mode_for_test = {
	if = {
		limit = {
			has_variable_map = cmm
			is_key_in_variable_map = {
				name = cmm
				target = flag:no_void_economy__nve_no_void_economy_main
			}
		}
		remove_from_variable_map = {
			name = cmm
			key = flag:no_void_economy__nve_no_void_economy_main
		}
	}
	add_to_variable_map = {
		name = cmm
		key = flag:no_void_economy__nve_no_void_economy_main
		value = $value$
	}
	modeu5_refresh_configuration_from_cmm_country_scope = yes
}

modeu5_perf14_restore_cmm_main_mode_after_test = {
	if = {
		limit = {
			has_variable_map = cmm
			is_key_in_variable_map = {
				name = cmm
				target = flag:no_void_economy__nve_no_void_economy_main
			}
		}
		remove_from_variable_map = {
			name = cmm
			key = flag:no_void_economy__nve_no_void_economy_main
		}
	}
	if = {
		limit = { scope:modeu5_perf14_original_cmm_main_present > 0 }
		add_to_variable_map = {
			name = cmm
			key = flag:no_void_economy__nve_no_void_economy_main
			value = scope:modeu5_perf14_original_cmm_main_value
		}
	}
	modeu5_refresh_configuration_from_cmm_country_scope = yes
}

modeu5_perf14_check_country_market_present = {
	$country$ = { save_temporary_scope_as = modeu5_perf14_presence_country }
	$market$ = { save_temporary_scope_as = modeu5_perf14_presence_market }

	save_temporary_scope_value_as = { name = modeu5_perf14_country_market_present value = 0 }

	scope:modeu5_perf14_presence_country = {
		set_local_variable = { name = modeu5_perf14_country_market_present value = 0 }
		every_market_present_in_country = {
			save_temporary_scope_as = modeu5_perf14_iterated_market
			if = {
				limit = { scope:modeu5_perf14_iterated_market = scope:modeu5_perf14_presence_market }
				scope:modeu5_perf14_presence_country = {
					set_local_variable = { name = modeu5_perf14_country_market_present value = 1 }
				}
			}
		}
		save_temporary_scope_value_as = {
			name = modeu5_perf14_country_market_present
			value = local_var:modeu5_perf14_country_market_present
		}
	}
}

modeu5_perf14_mirror_decision_to_ui = {
	if = {
		limit = { has_global_variable = modeu5_nve_main_mode }
		set_variable = {
			name = modeu5_perf14_ui_main_mode
			value = global_var:modeu5_nve_main_mode
		}
	}
	else = {
		set_variable = { name = modeu5_perf14_ui_main_mode value = 0 }
	}
	if = {
		limit = { modeu5_performance_mode_enabled_trigger = yes }
		set_variable = { name = modeu5_perf14_ui_performance_enabled value = 1 }
	}
	else = {
		set_variable = { name = modeu5_perf14_ui_performance_enabled value = 0 }
	}
	if = {
		limit = { modeu5_nve_normal_mode_trigger = yes }
		set_variable = { name = modeu5_perf14_ui_normal_enabled value = 1 }
	}
	else = {
		set_variable = { name = modeu5_perf14_ui_normal_enabled value = 0 }
	}
	if = {
		limit = { modeu5_nve_deactivated_trigger = yes }
		set_variable = { name = modeu5_perf14_ui_deactivated value = 1 }
	}
	else = {
		set_variable = { name = modeu5_perf14_ui_deactivated value = 0 }
	}
	set_variable = {
		name = modeu5_perf14_ui_detailed_enabled
		value = scope:modeu5_detailed_country_market_accounting_enabled
	}
	set_variable = {
		name = modeu5_perf14_ui_fallback_required
		value = scope:modeu5_market_level_fallback_required
	}
	set_variable = {
		name = modeu5_perf14_ui_country_is_human
		value = scope:modeu5_accounting_decision_country_is_human
	}
	set_variable = {
		name = modeu5_perf14_ui_market_human_relevant
		value = scope:modeu5_accounting_decision_market_human_relevant
	}
	if = {
		limit = { has_global_variable = modeu5_performance_relevant_market_added_count }
		set_variable = {
			name = modeu5_perf14_ui_human_relevant_market_count
			value = global_var:modeu5_performance_relevant_market_added_count
		}
	}
	else = {
		set_variable = { name = modeu5_perf14_ui_human_relevant_market_count value = 0 }
	}
	if = {
		limit = {
			has_global_variable_list = modeu5_performance_relevant_markets
			is_target_in_global_variable_list = {
				name = modeu5_performance_relevant_markets
				target = scope:modeu5_accounting_market
			}
		}
		set_variable = { name = modeu5_perf14_ui_current_market_human_relevant value = 1 }
	}
	else = {
		set_variable = { name = modeu5_perf14_ui_current_market_human_relevant value = 0 }
	}
}

modeu5_perf14_log_probe_result = {
	debug_log = "ModeU5 PERF-14 DUMP main_mode=[THIS.GetVariable('modeu5_perf14_ui_main_mode').GetValue|0] performance=[THIS.GetVariable('modeu5_perf14_ui_performance_enabled').GetValue|0] normal=[THIS.GetVariable('modeu5_perf14_ui_normal_enabled').GetValue|0] deactivated=[THIS.GetVariable('modeu5_perf14_ui_deactivated').GetValue|0] detailed=[THIS.GetVariable('modeu5_perf14_ui_detailed_enabled').GetValue|0] fallback=[THIS.GetVariable('modeu5_perf14_ui_fallback_required').GetValue|0] human=[THIS.GetVariable('modeu5_perf14_ui_country_is_human').GetValue|0] market_human_relevant=[THIS.GetVariable('modeu5_perf14_ui_market_human_relevant').GetValue|0] human_relevant_markets=[THIS.GetVariable('modeu5_perf14_ui_human_relevant_market_count').GetValue|0] current_market_human_relevant=[THIS.GetVariable('modeu5_perf14_ui_current_market_human_relevant').GetValue|0]"
	if = {
		limit = { has_global_variable = modeu5_test_perf14_performance_mode_cmm_failed }
		debug_log = "ModeU5 PERF-14 RESULT performance_mode_cmm FAIL"
		debug_log = "ModeU5 TEST FAIL scenario=perf14_performance_mode_cmm"
	}
	else_if = {
		limit = { has_global_variable = modeu5_test_perf14_performance_mode_cmm_passed }
		debug_log = "ModeU5 PERF-14 RESULT performance_mode_cmm PASS"
		debug_log = "ModeU5 TEST PASS scenario=perf14_performance_mode_cmm"
	}
	else_if = {
		limit = { has_global_variable = modeu5_test_perf14_performance_mode_cmm_blocked }
		debug_log = "ModeU5 PERF-14 RESULT performance_mode_cmm BLOCKED missing_fixture"
		debug_log = "ModeU5 TEST BLOCKED scenario=perf14_performance_mode_cmm"
	}
	else = {
		debug_log = "ModeU5 PERF-14 RESULT performance_mode_cmm PENDING"
		debug_log = "ModeU5 TEST PENDING scenario=perf14_performance_mode_cmm"
	}
}

modeu5_debug_run_perf14_performance_mode_cmm_test = {
	modeu5_perf14_clear_test_results = yes
	modeu5_enter_test_audit_runtime_mode = yes
	set_global_variable = modeu5_test_perf14_performance_mode_cmm_started
	debug_log = "ModeU5 TEST ENTERED scenario=perf14_performance_mode_cmm"

	save_temporary_scope_as = modeu5_perf14_test_country
	save_temporary_scope_value_as = { name = modeu5_perf14_original_cmm_main_present value = 0 }
	save_temporary_scope_value_as = { name = modeu5_perf14_original_cmm_main_value value = 1 }
	save_temporary_scope_value_as = { name = modeu5_perf14_ai_fixture_available value = 0 }
	save_temporary_scope_value_as = { name = modeu5_perf14_nonpresent_market_fixture_available value = 0 }

	if = {
		limit = {
			has_variable_map = cmm
			is_key_in_variable_map = {
				name = cmm
				target = flag:no_void_economy__nve_no_void_economy_main
			}
		}
		save_temporary_scope_value_as = { name = modeu5_perf14_original_cmm_main_present value = 1 }
		save_temporary_scope_value_as = {
			name = modeu5_perf14_original_cmm_main_value
			value = "variable_map(cmm|flag:no_void_economy__nve_no_void_economy_main)"
		}
	}

	if = {
		limit = { has_markets = no }
		set_global_variable = modeu5_test_perf14_performance_mode_cmm_blocked
		error_log = "ModeU5 PERF-14 performance mode CMM probe blocked because the current country has no markets."
	}
	else = {
		capital.market = { save_temporary_scope_as = modeu5_perf14_test_market }

		modeu5_perf14_set_cmm_main_mode_for_test = { value = 1 }
		# Force the non-detailed -> detailed edge case: the accounting gate must
		# mark a human-present market even when the monthly list is empty.
		modeu5_clear_performance_relevant_markets = yes
		modeu5_prepare_country_market_accounting_decision = {
			country = scope:modeu5_perf14_test_country
			market = scope:modeu5_perf14_test_market
		}
		modeu5_perf14_mirror_decision_to_ui = yes
		if = {
			limit = {
				OR = {
					NOT = { has_global_variable = modeu5_nve_main_mode }
					NOT = { global_var:modeu5_nve_main_mode = 1 }
					NOT = { modeu5_performance_mode_enabled_trigger = yes }
					NOT = { modeu5_detailed_country_market_accounting_enabled_trigger = yes }
					modeu5_market_level_fallback_required_trigger = yes
					NOT = { scope:modeu5_accounting_decision_country_is_human > 0 }
					NOT = { scope:modeu5_accounting_decision_market_human_relevant > 0 }
					NOT = {
						has_global_variable_list = modeu5_performance_relevant_markets
						is_target_in_global_variable_list = {
							name = modeu5_performance_relevant_markets
							target = scope:modeu5_perf14_test_market
						}
					}
				}
			}
			set_global_variable = modeu5_test_perf14_performance_mode_cmm_failed
		}

		if = {
			limit = {
				country_exists = c:ENG
				c:ENG = {
					is_ai = yes
					has_markets = yes
				}
			}
			save_temporary_scope_value_as = { name = modeu5_perf14_ai_fixture_available value = 1 }
			c:ENG = { save_temporary_scope_as = modeu5_perf14_ai_country }
			c:ENG.capital.market = { save_temporary_scope_as = modeu5_perf14_ai_market }
		}
		else_if = {
			limit = {
				country_exists = c:FRA
				c:FRA = {
					is_ai = yes
					has_markets = yes
				}
			}
			save_temporary_scope_value_as = { name = modeu5_perf14_ai_fixture_available value = 1 }
			c:FRA = { save_temporary_scope_as = modeu5_perf14_ai_country }
			c:FRA.capital.market = { save_temporary_scope_as = modeu5_perf14_ai_market }
		}
		else_if = {
			limit = {
				country_exists = c:POR
				c:POR = {
					is_ai = yes
					has_markets = yes
				}
			}
			save_temporary_scope_value_as = { name = modeu5_perf14_ai_fixture_available value = 1 }
			c:POR = { save_temporary_scope_as = modeu5_perf14_ai_country }
			c:POR.capital.market = { save_temporary_scope_as = modeu5_perf14_ai_market }
		}

		if = {
			limit = { scope:modeu5_perf14_ai_fixture_available > 0 }
			modeu5_clear_performance_relevant_markets = yes
			modeu5_prepare_country_market_accounting_decision = {
				country = scope:modeu5_perf14_ai_country
				market = scope:modeu5_perf14_ai_market
			}
			if = {
				limit = {
					OR = {
						modeu5_detailed_country_market_accounting_enabled_trigger = yes
						NOT = { modeu5_market_level_fallback_required_trigger = yes }
						scope:modeu5_accounting_decision_country_is_human > 0
					}
				}
				set_global_variable = modeu5_test_perf14_performance_mode_cmm_failed
			}
		}
		else = {
			set_global_variable = modeu5_test_perf14_performance_mode_cmm_blocked
			error_log = "ModeU5 PERF-14 performance mode CMM probe blocked because no AI country market fixture was available."
		}

		if = {
			limit = {
				country_exists = c:ENG
				c:ENG = { has_markets = yes }
				NOT = { c:ENG.capital.market = scope:modeu5_perf14_test_market }
			}
			c:ENG.capital.market = { save_temporary_scope_as = modeu5_perf14_nonpresent_market }
			modeu5_perf14_check_country_market_present = {
				country = scope:modeu5_perf14_test_country
				market = scope:modeu5_perf14_nonpresent_market
			}
			if = {
				limit = { scope:modeu5_perf14_country_market_present = 0 }
				save_temporary_scope_value_as = { name = modeu5_perf14_nonpresent_market_fixture_available value = 1 }
			}
		}
		if = {
			limit = {
				scope:modeu5_perf14_nonpresent_market_fixture_available = 0
				country_exists = c:FRA
				c:FRA = { has_markets = yes }
				NOT = { c:FRA.capital.market = scope:modeu5_perf14_test_market }
			}
			c:FRA.capital.market = { save_temporary_scope_as = modeu5_perf14_nonpresent_market }
			modeu5_perf14_check_country_market_present = {
				country = scope:modeu5_perf14_test_country
				market = scope:modeu5_perf14_nonpresent_market
			}
			if = {
				limit = { scope:modeu5_perf14_country_market_present = 0 }
				save_temporary_scope_value_as = { name = modeu5_perf14_nonpresent_market_fixture_available value = 1 }
			}
		}
		if = {
			limit = {
				scope:modeu5_perf14_nonpresent_market_fixture_available = 0
				country_exists = c:POR
				c:POR = { has_markets = yes }
				NOT = { c:POR.capital.market = scope:modeu5_perf14_test_market }
			}
			c:POR.capital.market = { save_temporary_scope_as = modeu5_perf14_nonpresent_market }
			modeu5_perf14_check_country_market_present = {
				country = scope:modeu5_perf14_test_country
				market = scope:modeu5_perf14_nonpresent_market
			}
			if = {
				limit = { scope:modeu5_perf14_country_market_present = 0 }
				save_temporary_scope_value_as = { name = modeu5_perf14_nonpresent_market_fixture_available value = 1 }
			}
		}

		if = {
			limit = { scope:modeu5_perf14_nonpresent_market_fixture_available > 0 }
			modeu5_clear_performance_relevant_markets = yes
			modeu5_prepare_country_market_accounting_decision = {
				country = scope:modeu5_perf14_test_country
				market = scope:modeu5_perf14_nonpresent_market
			}
			if = {
				limit = {
					OR = {
						modeu5_detailed_country_market_accounting_enabled_trigger = yes
						NOT = { modeu5_market_level_fallback_required_trigger = yes }
						NOT = { scope:modeu5_accounting_decision_country_is_human > 0 }
						scope:modeu5_accounting_decision_market_human_relevant > 0
					}
				}
				set_global_variable = modeu5_test_perf14_performance_mode_cmm_failed
			}
		}
		else = {
			set_global_variable = modeu5_test_perf14_performance_mode_cmm_blocked
			error_log = "ModeU5 PERF-14 performance mode CMM probe blocked because no non-present market fixture was available."
		}

		modeu5_perf14_set_cmm_main_mode_for_test = { value = 2 }
		modeu5_prepare_country_market_accounting_decision = {
			country = scope:modeu5_perf14_test_country
			market = scope:modeu5_perf14_test_market
		}
		if = {
			limit = {
				OR = {
					NOT = { has_global_variable = modeu5_nve_main_mode }
					NOT = { global_var:modeu5_nve_main_mode = 2 }
					NOT = { modeu5_nve_normal_mode_trigger = yes }
					NOT = { modeu5_detailed_country_market_accounting_enabled_trigger = yes }
					modeu5_market_level_fallback_required_trigger = yes
				}
			}
			set_global_variable = modeu5_test_perf14_performance_mode_cmm_failed
		}

		modeu5_perf14_set_cmm_main_mode_for_test = { value = 3 }
		modeu5_prepare_country_market_accounting_decision = {
			country = scope:modeu5_perf14_test_country
			market = scope:modeu5_perf14_test_market
		}
		if = {
			limit = {
				OR = {
					NOT = { has_global_variable = modeu5_nve_main_mode }
					NOT = { global_var:modeu5_nve_main_mode = 3 }
					NOT = { modeu5_nve_deactivated_trigger = yes }
					modeu5_detailed_country_market_accounting_enabled_trigger = yes
					modeu5_market_level_fallback_required_trigger = yes
				}
			}
			set_global_variable = modeu5_test_perf14_performance_mode_cmm_failed
		}

		if = {
			limit = {
				NOT = { has_global_variable = modeu5_test_perf14_performance_mode_cmm_failed }
				NOT = { has_global_variable = modeu5_test_perf14_performance_mode_cmm_blocked }
			}
			set_global_variable = modeu5_test_perf14_performance_mode_cmm_passed
		}
	}

	modeu5_perf14_restore_cmm_main_mode_after_test = yes
	modeu5_perf14_log_probe_result = yes
}
