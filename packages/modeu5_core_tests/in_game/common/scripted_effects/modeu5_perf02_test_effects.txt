# PERF-02 probes native country -> market traversal semantics.

modeu5_perf02_clear_probe_results = {
	modeu5_enter_audit_runtime_mode = yes

	remove_global_variable = modeu5_test_perf02_native_market_iterator_started
	remove_global_variable = modeu5_test_perf02_native_market_iterator_passed
	remove_global_variable = modeu5_test_perf02_native_market_iterator_failed
	remove_global_variable = modeu5_test_perf02_native_market_iterator_blocked
	remove_global_variable = modeu5_perf02_owned_location_market_count
	remove_global_variable = modeu5_perf02_owned_location_market_missing_count

	set_global_variable = { name = modeu5_perf02_owned_location_market_count value = 0 }
	set_global_variable = { name = modeu5_perf02_owned_location_market_missing_count value = 0 }

	if = {
		limit = { has_global_variable_list = modeu5_perf02_owned_location_markets }
		clear_global_variable_list = modeu5_perf02_owned_location_markets
	}

	remove_variable = modeu5_perf02_ui_human_country_count
	remove_variable = modeu5_perf02_ui_native_iterations
	remove_variable = modeu5_perf02_ui_relevant_markets
	remove_variable = modeu5_perf02_ui_owned_location_markets
	remove_variable = modeu5_perf02_ui_missing_owned_location_markets
}

modeu5_perf02_mark_owned_location_market = {
	if = {
		limit = {
			NOT = { has_global_variable_list = modeu5_perf02_owned_location_markets }
		}
		add_to_global_variable_list = {
			name = modeu5_perf02_owned_location_markets
			target = scope:modeu5_perf02_owned_location_market
		}
		set_global_variable = {
			name = modeu5_perf02_owned_location_market_count
			value = {
				value = global_var:modeu5_perf02_owned_location_market_count
				add = 1
			}
		}
	}
	else_if = {
		limit = {
			NOT = {
				is_target_in_global_variable_list = {
					name = modeu5_perf02_owned_location_markets
					target = scope:modeu5_perf02_owned_location_market
				}
			}
		}
		add_to_global_variable_list = {
			name = modeu5_perf02_owned_location_markets
			target = scope:modeu5_perf02_owned_location_market
		}
		set_global_variable = {
			name = modeu5_perf02_owned_location_market_count
			value = {
				value = global_var:modeu5_perf02_owned_location_market_count
				add = 1
			}
		}
	}
}

modeu5_perf02_mirror_probe_to_ui = {
	set_variable = {
		name = modeu5_perf02_ui_human_country_count
		value = global_var:modeu5_performance_human_country_count
	}
	set_variable = {
		name = modeu5_perf02_ui_native_iterations
		value = global_var:modeu5_performance_relevant_market_native_iterations
	}
	set_variable = {
		name = modeu5_perf02_ui_relevant_markets
		value = global_var:modeu5_performance_relevant_market_added_count
	}
	set_variable = {
		name = modeu5_perf02_ui_owned_location_markets
		value = global_var:modeu5_perf02_owned_location_market_count
	}
	set_variable = {
		name = modeu5_perf02_ui_missing_owned_location_markets
		value = global_var:modeu5_perf02_owned_location_market_missing_count
	}
}

modeu5_perf02_probe_current_country_native_markets = {
	modeu5_perf02_clear_probe_results = yes
	set_global_variable = modeu5_test_perf02_native_market_iterator_started
	save_temporary_scope_as = modeu5_performance_country

	modeu5_clear_performance_relevant_markets = yes
	set_global_variable = { name = modeu5_performance_human_country_count value = 1 }
	modeu5_add_country_present_markets_to_performance_relevant_list = yes

	if = {
		limit = { has_markets = no }
		set_global_variable = modeu5_test_perf02_native_market_iterator_blocked
	}
	else = {
		every_owned_location = {
			market = { save_temporary_scope_as = modeu5_perf02_owned_location_market }
			modeu5_perf02_mark_owned_location_market = yes
			if = {
				limit = { NOT = { has_global_variable_list = modeu5_performance_relevant_markets } }
				set_global_variable = {
					name = modeu5_perf02_owned_location_market_missing_count
					value = {
						value = global_var:modeu5_perf02_owned_location_market_missing_count
						add = 1
					}
				}
			}
			else_if = {
				limit = {
					NOT = {
						is_target_in_global_variable_list = {
							name = modeu5_performance_relevant_markets
							target = scope:modeu5_perf02_owned_location_market
						}
					}
				}
				set_global_variable = {
					name = modeu5_perf02_owned_location_market_missing_count
					value = {
						value = global_var:modeu5_perf02_owned_location_market_missing_count
						add = 1
					}
				}
			}
		}
		if = {
			limit = { global_var:modeu5_perf02_owned_location_market_missing_count > 0 }
			set_global_variable = modeu5_test_perf02_native_market_iterator_failed
			error_log = "ModeU5 PERF-02 native market iterator probe found an owned-location market missing from every_market_present_in_country."
		}
		else = {
			set_global_variable = modeu5_test_perf02_native_market_iterator_passed
		}
	}

	modeu5_perf02_mirror_probe_to_ui = yes
}
