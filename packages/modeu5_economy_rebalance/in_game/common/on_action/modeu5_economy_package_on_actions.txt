# Snapshot package presence for a new campaign. Existing-save package changes are unsupported.
on_game_start = {
	on_actions = {
		modeu5_economy_rebalance_package_on_game_start
	}
}

modeu5_economy_rebalance_package_on_game_start = {
	effect = {
		set_global_variable = modeu5_economy_rebalance_loaded
		set_global_variable = {
			name = modeu5_economy_package_version
			value = 100
		}
	}
}
