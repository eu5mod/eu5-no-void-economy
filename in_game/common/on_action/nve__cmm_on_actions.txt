# Hook this mod into CMF shared registration on_action.
cmf_on_mod_registration = {
	on_actions = {
		nve__on_register_cmf_mod
	}
}

nve__on_register_cmf_mod = {
	effect = {
		nve__register_cmf_mod = yes
	}
}

# Unified callback hook for setting changes, alert clicks, action bar clicks.
cmf_on_callback = {
	on_actions = {
		nve__on_cmf_callback
	}
}

nve__on_cmf_callback = {
	effect = {
		nve__handle_cmf_callback = yes
	}
}
