on_location_changed_owner = {
	on_actions = {
		modeu5_core03_probe_location_changed_owner
	}
}

modeu5_core03_probe_location_changed_owner = {
	effect = {
		modeu5_core03_handle_location_changed_owner = yes
		modeu5_core03_probe_record_location_changed_owner = yes
	}
}

	}
}

	effect = {
		modeu5_core03_probe_record_new_country_formed = yes
		modeu5_core03_handle_new_country_finalizer = yes
