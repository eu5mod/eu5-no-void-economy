# Callback handler for cmf_on_callback.
# var:cmf_callback is the flag of the setting, alert, or action bar element that was interacted with.
# Scope: country
nve__handle_cmf_callback = {
	switch = {
		trigger = var:cmf_callback
		# Economic Rework (nve_general) Tab
		## No Void Economy (nve_no_void_economy) Group
		flag:no_void_economy__nve_no_void_economy_main = {
			cmm_sync_dropdown_option_alias = {
				setting = no_void_economy__nve_no_void_economy_main
				index = 3
				alias = variable_mapcmmflagno_void_economy__activate_no_void_economy0
			}
			nve_cmf_activate_no_void_economy = yes
		}
		flag:no_void_economy__nve_decay_activate = {
			nve_cmf_activate_good_decay = yes
		}
	}
}
