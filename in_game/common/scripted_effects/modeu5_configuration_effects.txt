# ModeU5 package selection is owned by the launcher playset.
# Community Mod Framework configures diagnostic/performance behavior.
# Package presence is captured for a new campaign; existing-save package changes are unsupported.

modeu5_clear_runtime_mode_flags = {
	remove_global_variable = modeu5_runtime_mode_normal
	remove_global_variable = modeu5_runtime_mode_debug
	remove_global_variable = modeu5_runtime_mode_audit
}

modeu5_enter_normal_runtime_mode = {
	modeu5_clear_runtime_mode_flags = yes
	set_global_variable = modeu5_runtime_mode_normal
}

modeu5_enter_debug_runtime_mode = {
	modeu5_clear_runtime_mode_flags = yes
	set_global_variable = modeu5_runtime_mode_debug
}

modeu5_enter_audit_runtime_mode = {
	modeu5_clear_runtime_mode_flags = yes
	set_global_variable = modeu5_runtime_mode_audit
}

modeu5_enter_test_audit_runtime_mode = {
	modeu5_clear_runtime_mode_flags = yes
	set_global_variable = modeu5_runtime_mode_debug
	set_global_variable = modeu5_runtime_mode_audit
}

modeu5_clear_accounting_persistence_flags = {
	remove_global_variable = modeu5_accounting_persistence_minimal
	remove_global_variable = modeu5_accounting_persistence_human_relevant
	remove_global_variable = modeu5_accounting_persistence_strict
}

modeu5_enter_minimal_accounting_persistence = {
	modeu5_clear_accounting_persistence_flags = yes
	set_global_variable = modeu5_accounting_persistence_minimal
}

modeu5_enter_strict_accounting_persistence = {
	modeu5_clear_accounting_persistence_flags = yes
	set_global_variable = modeu5_accounting_persistence_strict
}

modeu5_enter_human_relevant_accounting_persistence = {
	modeu5_clear_accounting_persistence_flags = yes
	set_global_variable = modeu5_accounting_persistence_human_relevant
}

modeu5_apply_debug_mode_cmm_defaults = {
	if = {
		limit = { debug_only = yes }
		if = {
			limit = {
				has_variable_map = cmm
				is_key_in_variable_map = {
					name = cmm
					target = flag:no_void_economy__nve_debug_messages
				}
			}
			remove_from_variable_map = {
				name = cmm
				key = flag:no_void_economy__nve_debug_messages
			}
		}
		add_to_variable_map = {
			name = cmm
			key = flag:no_void_economy__nve_debug_messages
			value = 3
		}
	}
}

modeu5_initialize_configuration_state_effect = {
	if = {
		limit = {
			NOT = { has_global_variable = modeu5_core_package_loaded }
		}
		set_global_variable = modeu5_core_package_loaded
	}
	set_global_variable = {
		name = modeu5_core_package_version
		value = 100
	}

	modeu5_enter_normal_runtime_mode = yes
	set_global_variable = {
		name = modeu5_debug_level
		value = 0
	}
	modeu5_enter_minimal_accounting_persistence = yes
	set_global_variable = modeu5_cmm_nve_main_enabled
	modeu5_apply_debug_mode_cmm_defaults = yes
	modeu5_refresh_configuration_from_cmm_country_scope = yes
}

modeu5_refresh_configuration_from_cmm_country_scope = {
	modeu5_apply_debug_mode_cmm_defaults = yes

	if = {
		limit = {
			has_variable_map = cmm
			is_key_in_variable_map = {
				name = cmm
				target = flag:no_void_economy__nve_debug_messages
			}
			"variable_map(cmm|flag:no_void_economy__nve_debug_messages)" = 3
		}
		modeu5_enter_debug_runtime_mode = yes
		set_global_variable = {
			name = modeu5_debug_level
			value = 2
		}
	}
	else_if = {
		limit = {
			has_variable_map = cmm
			is_key_in_variable_map = {
				name = cmm
				target = flag:no_void_economy__nve_debug_messages
			}
			"variable_map(cmm|flag:no_void_economy__nve_debug_messages)" = 2
		}
		modeu5_enter_debug_runtime_mode = yes
		set_global_variable = {
			name = modeu5_debug_level
			value = 1
		}
	}
	else = {
		modeu5_enter_normal_runtime_mode = yes
		set_global_variable = {
			name = modeu5_debug_level
			value = 0
		}
	}

	if = {
		limit = {
			has_variable_map = cmm
			is_key_in_variable_map = {
				name = cmm
				target = flag:no_void_economy__nve_monthly_stock_check
			}
			"variable_map(cmm|flag:no_void_economy__nve_monthly_stock_check)" = 2
		}
		if = {
			limit = { modeu5_debug_enabled_trigger = yes }
			modeu5_enter_test_audit_runtime_mode = yes
		}
		else = {
			modeu5_enter_audit_runtime_mode = yes
		}
	}

	if = {
		limit = {
			has_variable_map = cmm
			is_key_in_variable_map = {
				name = cmm
				target = flag:no_void_economy__nve_save_mode
			}
			"variable_map(cmm|flag:no_void_economy__nve_save_mode)" = 3
		}
		modeu5_enter_strict_accounting_persistence = yes
	}
	else_if = {
		limit = {
			has_variable_map = cmm
			is_key_in_variable_map = {
				name = cmm
				target = flag:no_void_economy__nve_save_mode
			}
			"variable_map(cmm|flag:no_void_economy__nve_save_mode)" = 2
		}
		modeu5_enter_human_relevant_accounting_persistence = yes
	}
	else = {
		modeu5_enter_minimal_accounting_persistence = yes
	}
}
