# ModeU5 package selection is owned by the launcher playset.
# The built-in game-rule screen configures diagnostic behavior only.
# Package presence is captured for a new campaign; existing-save package changes are unsupported.

modeu5_clear_runtime_mode_flags = {
	remove_global_variable = modeu5_runtime_mode_normal
	remove_global_variable = modeu5_runtime_mode_debug
	remove_global_variable = modeu5_runtime_mode_audit
}

modeu5_enter_normal_runtime_mode = {
	modeu5_clear_runtime_mode_flags = yes
	set_global_variable = modeu5_runtime_mode_normal
}

modeu5_enter_debug_runtime_mode = {
	modeu5_clear_runtime_mode_flags = yes
	set_global_variable = modeu5_runtime_mode_debug
}

modeu5_enter_audit_runtime_mode = {
	modeu5_clear_runtime_mode_flags = yes
	set_global_variable = modeu5_runtime_mode_audit
}

modeu5_enter_test_audit_runtime_mode = {
	modeu5_clear_runtime_mode_flags = yes
	set_global_variable = modeu5_runtime_mode_debug
	set_global_variable = modeu5_runtime_mode_audit
}

modeu5_clear_accounting_persistence_flags = {
	remove_global_variable = modeu5_accounting_persistence_minimal
	remove_global_variable = modeu5_accounting_persistence_strict
}

modeu5_enter_minimal_accounting_persistence = {
	modeu5_clear_accounting_persistence_flags = yes
	set_global_variable = modeu5_accounting_persistence_minimal
}

modeu5_enter_strict_accounting_persistence = {
	modeu5_clear_accounting_persistence_flags = yes
	set_global_variable = modeu5_accounting_persistence_strict
}

modeu5_initialize_configuration_state_effect = {
	if = {
		limit = {
			NOT = { has_global_variable = modeu5_core_package_loaded }
		}
		set_global_variable = modeu5_core_package_loaded
	}
	set_global_variable = {
		name = modeu5_core_package_version
		value = 100
	}

	if = {
		limit = { has_game_rule = modeu5_debug_verbose }
		modeu5_enter_debug_runtime_mode = yes
		set_global_variable = {
			name = modeu5_debug_level
			value = 2
		}
	}
	else_if = {
		limit = { has_game_rule = modeu5_debug_basic }
		modeu5_enter_debug_runtime_mode = yes
		set_global_variable = {
			name = modeu5_debug_level
			value = 1
		}
	}
	else = {
		modeu5_enter_normal_runtime_mode = yes
		set_global_variable = {
			name = modeu5_debug_level
			value = 0
		}
	}

	if = {
		limit = { has_game_rule = modeu5_audit_on }
		modeu5_enter_audit_runtime_mode = yes
	}

	if = {
		limit = { has_game_rule = modeu5_accounting_strict }
		modeu5_enter_strict_accounting_persistence = yes
	}
	else = {
		modeu5_enter_minimal_accounting_persistence = yes
	}
}
