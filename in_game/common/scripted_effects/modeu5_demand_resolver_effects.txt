# ModeU5 US-10 demand resolution effects.
#
# These effects intentionally implement the explicit-request MVP. Runtime
# vanilla Pop demand and exact vanilla trade quantities remain partially
# unconfirmed, so callers must pass the requested quantity and the stock source
# they want to consume or transfer. Stock mutation still goes exclusively
# through CORE-01 centralized operators.

modeu5_select_demand_type_consumption = {
	save_temporary_scope_value_as = {
		name = modeu5_demand_type
		value = 1
	}
}

modeu5_select_demand_type_inter_market_trade = {
	save_temporary_scope_value_as = {
		name = modeu5_demand_type
		value = 2
	}
}

modeu5_record_country_market_demand_outcome = {
	$country$ = { save_temporary_scope_as = modeu5_us10_outcome_country }
	$market$ = { save_temporary_scope_as = modeu5_market }
	goods:$good$ = { save_temporary_scope_as = modeu5_good }

	save_temporary_scope_value_as = {
		name = modeu5_us10_requested_quantity_raw
		value = $requested_quantity$
	}
	save_temporary_scope_value_as = {
		name = modeu5_us10_requested_quantity
		value = {
			value = scope:modeu5_us10_requested_quantity_raw
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_us10_satisfied_quantity
		value = {
			value = $satisfied_quantity$
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_us10_transferred_quantity
		value = {
			value = $transferred_quantity$
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_us10_unsatisfied_quantity
		value = {
			value = $unsatisfied_quantity$
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_us10_satisfaction_ratio
		value = 0
	}
	if = {
		limit = { scope:modeu5_us10_requested_quantity > 0 }
		save_temporary_scope_value_as = {
			name = modeu5_us10_satisfaction_ratio
			value = {
				value = scope:modeu5_us10_satisfied_quantity
				divide = scope:modeu5_us10_requested_quantity
			}
		}
	}

	modeu5_select_demand_type_$demand_type$ = yes

	if = {
		limit = { scope:modeu5_us10_requested_quantity > 0 }
		modeu5_store_us10_country_market_outcome_good_$good$ = yes
	}
}

modeu5_clear_country_market_demand_outcome = {
	$country$ = { save_temporary_scope_as = modeu5_us10_outcome_country }
	$market$ = { save_temporary_scope_as = modeu5_market }
	goods:$good$ = { save_temporary_scope_as = modeu5_good }

	modeu5_clear_us10_country_market_outcome_good_$good$ = yes
}

modeu5_debug_capture_us10_demand_resolution = {
	if = {
		limit = { modeu5_debug_capture_enabled_trigger = yes }
		$country$ = {
			remove_variable = modeu5_debug_us10_last_demand_type
			remove_variable = modeu5_debug_us10_last_good
			remove_variable = modeu5_debug_us10_last_market
			remove_variable = modeu5_debug_us10_last_source_market
			remove_variable = modeu5_debug_us10_last_target_market
			remove_variable = modeu5_debug_us10_last_provider_country
			remove_variable = modeu5_debug_us10_last_buyer_country
			remove_variable = modeu5_debug_us10_last_seller_country
			remove_variable = modeu5_debug_us10_last_requested_quantity
			remove_variable = modeu5_debug_us10_last_satisfied_quantity
			remove_variable = modeu5_debug_us10_last_transferred_quantity
			remove_variable = modeu5_debug_us10_last_unsatisfied_quantity
			remove_variable = modeu5_debug_us10_last_satisfaction_ratio
			remove_variable = modeu5_debug_us10_last_buyer_available_capacity
			remove_variable = modeu5_debug_us10_last_is_intra_market_trade
			remove_variable = modeu5_debug_us10_last_trade_income_generated
			remove_variable = modeu5_debug_us10_last_transport_cost_generated
			remove_variable = modeu5_debug_us10_last_trade_capacity_used

			set_variable = { name = modeu5_debug_us10_last_demand_type value = scope:modeu5_demand_type }
			set_variable = { name = modeu5_debug_us10_last_good value = scope:modeu5_good }
			if = {
				limit = { exists = scope:modeu5_market }
				set_variable = { name = modeu5_debug_us10_last_market value = scope:modeu5_market }
			}
			if = {
				limit = { exists = scope:modeu5_source_market }
				set_variable = { name = modeu5_debug_us10_last_source_market value = scope:modeu5_source_market }
			}
			if = {
				limit = { exists = scope:modeu5_target_market }
				set_variable = { name = modeu5_debug_us10_last_target_market value = scope:modeu5_target_market }
			}
			if = {
				limit = { exists = scope:modeu5_provider_country }
				set_variable = { name = modeu5_debug_us10_last_provider_country value = scope:modeu5_provider_country }
			}
			if = {
				limit = { exists = scope:modeu5_buyer_country }
				set_variable = { name = modeu5_debug_us10_last_buyer_country value = scope:modeu5_buyer_country }
			}
			if = {
				limit = { exists = scope:modeu5_seller_country }
				set_variable = { name = modeu5_debug_us10_last_seller_country value = scope:modeu5_seller_country }
			}
			set_variable = { name = modeu5_debug_us10_last_requested_quantity value = scope:modeu5_us10_requested_quantity }
			set_variable = { name = modeu5_debug_us10_last_satisfied_quantity value = scope:modeu5_us10_satisfied_quantity }
			set_variable = { name = modeu5_debug_us10_last_transferred_quantity value = scope:modeu5_us10_transferred_quantity }
			set_variable = { name = modeu5_debug_us10_last_unsatisfied_quantity value = scope:modeu5_us10_unsatisfied_quantity }
			set_variable = { name = modeu5_debug_us10_last_satisfaction_ratio value = scope:modeu5_us10_satisfaction_ratio }
			set_variable = { name = modeu5_debug_us10_last_buyer_available_capacity value = scope:modeu5_us10_buyer_available_capacity }
			set_variable = { name = modeu5_debug_us10_last_is_intra_market_trade value = 0 }
			set_variable = { name = modeu5_debug_us10_last_trade_income_generated value = 0 }
			set_variable = { name = modeu5_debug_us10_last_transport_cost_generated value = 0 }
			set_variable = { name = modeu5_debug_us10_last_trade_capacity_used value = 0 }
		}
	}
}

modeu5_resolve_consumption_stock_demand = {
	$demanding_country$ = { save_temporary_scope_as = modeu5_demanding_country }
	$provider_country$ = { save_temporary_scope_as = modeu5_provider_country }
	$market$ = { save_temporary_scope_as = modeu5_market }
	goods:$good$ = { save_temporary_scope_as = modeu5_good }

	save_temporary_scope_value_as = {
		name = modeu5_us10_requested_quantity_raw
		value = $requested_quantity$
	}
	save_temporary_scope_value_as = {
		name = modeu5_us10_requested_quantity
		value = {
			value = scope:modeu5_us10_requested_quantity_raw
			min = 0
		}
	}
	save_temporary_scope_value_as = { name = modeu5_us10_satisfied_quantity value = 0 }
	save_temporary_scope_value_as = { name = modeu5_us10_transferred_quantity value = 0 }
	save_temporary_scope_value_as = { name = modeu5_us10_unsatisfied_quantity value = scope:modeu5_us10_requested_quantity }
	save_temporary_scope_value_as = { name = modeu5_us10_buyer_available_capacity value = 0 }
	modeu5_select_demand_type_consumption = yes

	if = {
		limit = { scope:modeu5_us10_requested_quantity > 0 }
		modeu5_remove_stock = {
			country = scope:modeu5_provider_country
			market = scope:modeu5_market
			good = $good$
			quantity = scope:modeu5_us10_requested_quantity
			reason = consumption
		}
		save_temporary_scope_value_as = {
			name = modeu5_us10_satisfied_quantity
			value = scope:modeu5_actual_removed_quantity
		}
		save_temporary_scope_value_as = {
			name = modeu5_us10_unsatisfied_quantity
			value = scope:modeu5_unsatisfied_quantity
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_us10_satisfaction_ratio
		value = 0
	}
	if = {
		limit = { scope:modeu5_us10_requested_quantity > 0 }
		save_temporary_scope_value_as = {
			name = modeu5_us10_satisfaction_ratio
			value = {
				value = scope:modeu5_us10_satisfied_quantity
				divide = scope:modeu5_us10_requested_quantity
			}
		}
	}

	modeu5_record_country_market_demand_outcome = {
		country = scope:modeu5_demanding_country
		market = scope:modeu5_market
		good = $good$
		demand_type = consumption
		requested_quantity = scope:modeu5_us10_requested_quantity
		satisfied_quantity = scope:modeu5_us10_satisfied_quantity
		transferred_quantity = 0
		unsatisfied_quantity = scope:modeu5_us10_unsatisfied_quantity
	}
	modeu5_debug_capture_us10_demand_resolution = {
		country = scope:modeu5_demanding_country
	}
}

modeu5_resolve_inter_market_stock_transfer = {
	$buyer_country$ = { save_temporary_scope_as = modeu5_buyer_country }
	$seller_country$ = { save_temporary_scope_as = modeu5_seller_country }
	$source_market$ = { save_temporary_scope_as = modeu5_source_market }
	$target_market$ = { save_temporary_scope_as = modeu5_target_market }
	goods:$good$ = { save_temporary_scope_as = modeu5_good }

	save_temporary_scope_value_as = {
		name = modeu5_us10_requested_quantity_raw
		value = $requested_quantity$
	}
	save_temporary_scope_value_as = {
		name = modeu5_us10_requested_quantity
		value = {
			value = scope:modeu5_us10_requested_quantity_raw
			min = 0
		}
	}
	save_temporary_scope_value_as = { name = modeu5_us10_satisfied_quantity value = 0 }
	save_temporary_scope_value_as = { name = modeu5_us10_transferred_quantity value = 0 }
	save_temporary_scope_value_as = { name = modeu5_us10_unsatisfied_quantity value = scope:modeu5_us10_requested_quantity }
	save_temporary_scope_value_as = { name = modeu5_us10_buyer_available_capacity value = 0 }
	modeu5_select_demand_type_inter_market_trade = yes

	modeu5_read_country_stock_record = {
		country = scope:modeu5_buyer_country
		market = scope:modeu5_target_market
		good = $good$
	}
	save_temporary_scope_value_as = {
		name = modeu5_us10_buyer_available_capacity
		value = scope:modeu5_read_available_capacity
	}

	if = {
		limit = {
			scope:modeu5_us10_requested_quantity > 0
			scope:modeu5_source_market != scope:modeu5_target_market
		}
		modeu5_transfer_stock = {
			seller_country = scope:modeu5_seller_country
			buyer_country = scope:modeu5_buyer_country
			source_market = scope:modeu5_source_market
			target_market = scope:modeu5_target_market
			good = $good$
			quantity = scope:modeu5_us10_requested_quantity
			target_capacity_policy = enforce
		}
		save_temporary_scope_value_as = {
			name = modeu5_us10_satisfied_quantity
			value = scope:modeu5_actual_transferred_quantity
		}
		save_temporary_scope_value_as = {
			name = modeu5_us10_transferred_quantity
			value = scope:modeu5_actual_transferred_quantity
		}
		save_temporary_scope_value_as = {
			name = modeu5_us10_unsatisfied_quantity
			value = scope:modeu5_unsatisfied_quantity
		}
		save_temporary_scope_value_as = {
			name = modeu5_us10_buyer_available_capacity
			value = scope:modeu5_available_capacity
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_us10_satisfaction_ratio
		value = 0
	}
	if = {
		limit = { scope:modeu5_us10_requested_quantity > 0 }
		save_temporary_scope_value_as = {
			name = modeu5_us10_satisfaction_ratio
			value = {
				value = scope:modeu5_us10_satisfied_quantity
				divide = scope:modeu5_us10_requested_quantity
			}
		}
	}

	modeu5_record_country_market_demand_outcome = {
		country = scope:modeu5_buyer_country
		market = scope:modeu5_target_market
		good = $good$
		demand_type = inter_market_trade
		requested_quantity = scope:modeu5_us10_requested_quantity
		satisfied_quantity = scope:modeu5_us10_satisfied_quantity
		transferred_quantity = scope:modeu5_us10_transferred_quantity
		unsatisfied_quantity = scope:modeu5_us10_unsatisfied_quantity
	}
	modeu5_debug_capture_us10_demand_resolution = {
		country = scope:modeu5_buyer_country
	}
}
