modeu5_core03_probe_record_location_changed_owner = {
	modeu5_core03_probe_increment_sequence = yes
	modeu5_core03_probe_increment_location_count = yes
	set_global_variable = modeu5_core03_probe_location_hook_observed
	set_global_variable = {
		name = modeu5_core03_probe_last_location_sequence
		value = global_var:modeu5_core03_probe_sequence
	}
	error_log = "ModeU5 CORE-03 OBSERVED on_location_changed_owner."

	if = {
		limit = {
			exists = scope:loser
			exists = scope:winner
			scope:loser != scope:winner
		}
		set_global_variable = modeu5_core03_probe_location_scopes_valid
		error_log = "ModeU5 CORE-03 PASS location hook resolved distinct loser and winner scopes."
	}
	else = {
		set_global_variable = modeu5_core03_probe_location_scopes_invalid
		error_log = "ModeU5 CORE-03 FAIL location hook did not resolve distinct loser and winner scopes."
	}

	save_temporary_scope_as = modeu5_core03_probe_location
	if = {
		limit = {
			NOT = {
				has_global_variable_list = modeu5_core03_probe_seen_locations
			}
		}
		add_to_global_variable_list = {
			name = modeu5_core03_probe_seen_locations
			target = scope:modeu5_core03_probe_location
		}
	}
	else_if = {
		limit = {
			NOT = {
				is_target_in_global_variable_list = {
					name = modeu5_core03_probe_seen_locations
					target = scope:modeu5_core03_probe_location
				}
			}
		}
		add_to_global_variable_list = {
			name = modeu5_core03_probe_seen_locations
			target = scope:modeu5_core03_probe_location
		}
	}
	else = {
		set_global_variable = modeu5_core03_probe_duplicate_location_hook
		error_log = "ModeU5 CORE-03 FAIL duplicate on_location_changed_owner observed for the same location."
	}
}

	if = {
		limit = {
			country_exists = c:CAS
			location:leon = {
				owner ?= c:CAS
			}
			c:CAS = {

						capital ?= location:leon
					}

				set_capital = prev
				make_subject_of = { target = c:CAS type = subject_type:vassal }

modeu5_core03_probe_execute_staged_annexation = {
	if = {
		limit = {
			has_global_variable = modeu5_core03_probe_annexation_stage_ready
			country_exists = c:CAS
			c:CAS = {
				any_subject = {
					capital ?= location:leon
				}
			}
		}
		modeu5_core03_probe_select_scenario_annexation = yes
		set_global_variable = modeu5_core03_probe_synthetic_run
		set_global_variable = modeu5_core03_probe_synthetic_runner_started
		error_log = "ModeU5 CORE-03 scripted Scenario C annexation step started: Castile will annex the staged Leon subject."
		c:CAS = {
			every_subject = {
				limit = {
					capital ?= location:leon
				}
				save_temporary_scope_as = modeu5_core03_probe_target_country
			}
		}
		c:CAS = {
			annex_country = {
				country = scope:modeu5_core03_probe_target_country
				reason = Diplomatic
				transfer = yes
			}
		}
		set_global_variable = modeu5_core03_probe_synthetic_runner_completed
		error_log = "ModeU5 CORE-03 scripted Scenario C completed: the staged Leon subject should now be annexed by Castile."
		c:CAS = {
			trigger_event_non_silently = { id = modeu5_core03_probe.2 days = 2 }
		}
	}
	else = {
		set_global_variable = modeu5_core03_probe_synthetic_fixture_invalid
		set_global_variable = modeu5_core03_probe_synthetic_runner_aborted
		error_log = "ModeU5 CORE-03 scripted Scenario C annexation step aborted because the staged Leon subject could not be found."
