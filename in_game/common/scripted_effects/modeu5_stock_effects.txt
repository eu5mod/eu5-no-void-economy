# ModeU5 centralized stock mutation effects.
#
# Public calls use a literal goods token so each expanded map name remains static:
#
# modeu5_add_stock = {
#     country = root
#     market = root.capital.market
#     good = wheat
#     quantity = 50
#     capacity_policy = enforce
# }
#
# Supported capacity policies are enforce and allow_over_capacity. The latter is
# reserved for CORE-02, CORE-03, and separately approved migrations.

modeu5_select_capacity_policy_enforce = {
	save_temporary_scope_value_as = {
		name = modeu5_capacity_policy
		value = 0
	}
}

	save_temporary_scope_value_as = {
		name = modeu5_capacity_policy

# Generated per-good adapters read and replace literal physical map names.
# Shared effects below only calculate results from temporary inputs; map
# identifiers are never passed through scripted-effect parameters.

# US-01 read-only country x market x good record access.

modeu5_read_country_stock_record = {
	$country$ = { save_temporary_scope_as = modeu5_country }
	$market$ = { save_temporary_scope_as = modeu5_market }
	goods:$good$ = { save_temporary_scope_as = modeu5_good }

	modeu5_read_country_stock_record_good_$good$ = yes
}

modeu5_read_country_stock_record_calculate = {
	save_temporary_scope_value_as = {
		name = modeu5_read_country_stock
		value = {
			value = scope:modeu5_country_stock_before_raw
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_read_country_stock_cap
		value = {
			value = scope:modeu5_country_stock_cap_raw
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_read_country_base_capacity
		value = {
			value = scope:modeu5_country_base_capacity_raw
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_read_country_building_capacity
		value = {
			value = scope:modeu5_country_building_capacity_raw
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_read_country_foreign_capacity
		value = {
			value = scope:modeu5_country_foreign_capacity_raw
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_read_available_capacity
		value = {
			value = scope:modeu5_read_country_stock_cap
			subtract = scope:modeu5_read_country_stock
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_read_over_capacity
		value = {
			value = scope:modeu5_read_country_stock
			subtract = scope:modeu5_read_country_stock_cap
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_read_market_stock
		value = {
			value = scope:modeu5_market_stock_before
			min = 0
		}
	}
}

# CORE-01.1

modeu5_add_stock = {
	$country$ = { save_temporary_scope_as = modeu5_country }
	$market$ = { save_temporary_scope_as = modeu5_market }
	goods:$good$ = { save_temporary_scope_as = modeu5_good }

	save_temporary_scope_value_as = {
		name = modeu5_quantity_requested_raw
		value = $quantity$
	}
	save_temporary_scope_value_as = {
		name = modeu5_quantity_requested
		value = {
			value = scope:modeu5_quantity_requested_raw
			min = 0
		}
	}
	if = {
		limit = { scope:modeu5_quantity_requested_raw < 0 }
		error_log = "ModeU5 modeu5_add_stock received a negative quantity; request normalized to zero."
	}

	modeu5_select_capacity_policy_$capacity_policy$ = yes
	modeu5_add_stock_good_$good$ = yes
}

modeu5_add_stock_calculate = {
	save_temporary_scope_value_as = {
		name = modeu5_country_stock_before
		value = {
			value = scope:modeu5_country_stock_before_raw
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_country_stock_cap
		value = {
			value = scope:modeu5_country_stock_cap_raw
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_available_capacity
		value = {
			value = scope:modeu5_country_stock_cap
			subtract = scope:modeu5_country_stock_before
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_over_capacity_before
		value = {
			value = scope:modeu5_country_stock_before
			subtract = scope:modeu5_country_stock_cap
			min = 0
		}
	}

	if = {
		limit = { scope:modeu5_capacity_policy = 1 }
		save_temporary_scope_value_as = {
			name = modeu5_actual_added_quantity
			value = scope:modeu5_quantity_requested
		}
	}
	else = {
		save_temporary_scope_value_as = {
			name = modeu5_actual_added_quantity
			value = {
				value = scope:modeu5_quantity_requested
				max = scope:modeu5_available_capacity
			}
		}
	}

	save_temporary_scope_value_as = {
		name = modeu5_rejected_quantity
		value = {
			value = scope:modeu5_quantity_requested
			subtract = scope:modeu5_actual_added_quantity
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_country_stock_after
		value = {
			value = scope:modeu5_country_stock_before
			add = scope:modeu5_actual_added_quantity
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_market_stock_after
		value = {
			value = scope:modeu5_market_stock_before
			add = scope:modeu5_actual_added_quantity
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_over_capacity_after
		value = {
			value = scope:modeu5_country_stock_after
			subtract = scope:modeu5_country_stock_cap
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_over_capacity_created
		value = {
			value = scope:modeu5_over_capacity_after
			subtract = scope:modeu5_over_capacity_before
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_stock_difference_after
		value = {
			value = scope:modeu5_market_stock_after
			subtract = scope:modeu5_market_stock_before
			subtract = scope:modeu5_country_stock_after
			add = scope:modeu5_country_stock_before
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_consistency_validation_required
		value = 0
	}

	if = {
		limit = { scope:modeu5_country_stock_before_raw < 0 }
		error_log = "ModeU5 modeu5_add_stock found a negative country stock; the country record is normalized and consistency validation is required."
		save_temporary_scope_value_as = {
			name = modeu5_consistency_validation_required
			value = 1
		}
	}
	if = {
		limit = { scope:modeu5_country_stock_cap_raw < 0 }
		error_log = "ModeU5 modeu5_add_stock found a negative capacity; capacity is treated as zero."
	}
	if = {
		limit = { scope:modeu5_market_stock_before < 0 }
		error_log = "ModeU5 modeu5_add_stock found a negative market aggregate; consistency validation is required."
		save_temporary_scope_value_as = {
			name = modeu5_consistency_validation_required
			value = 1
		}
	}

	save_temporary_scope_value_as = {
		name = modeu5_country_record_write_required
		value = 0
	}
	if = {
		limit = {
			OR = {
				scope:modeu5_actual_added_quantity > 0
				scope:modeu5_country_stock_before_raw < 0
			}
		}
		save_temporary_scope_value_as = {
			name = modeu5_country_record_write_required
			value = 1
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_market_record_write_required
		value = 0
	}
	if = {
		limit = { scope:modeu5_actual_added_quantity > 0 }
		save_temporary_scope_value_as = {
			name = modeu5_market_record_write_required
			value = 1
		}
	}

}

modeu5_remove_stock_calculate = {
	save_temporary_scope_value_as = {
		name = modeu5_country_stock_before
		value = {
			value = scope:modeu5_country_stock_before_raw
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_country_stock_cap
		value = {
			value = scope:modeu5_country_stock_cap_raw
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_actual_removed_quantity
		value = {
			value = scope:modeu5_quantity_requested
			max = scope:modeu5_country_stock_before
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_consistency_validation_required
		value = 0
	}

	# Precision note: a tiny aggregate residue could trip this strict underflow
	# guard. Rebuild and inspect raw values before considering any tolerance.
	if = {
		limit = {
			OR = {
				scope:modeu5_market_stock_before < 0
				scope:modeu5_market_stock_before < scope:modeu5_actual_removed_quantity
			}
		}
		error_log = "ModeU5 modeu5_remove_stock blocked a market aggregate underflow; consistency validation is required."
		save_temporary_scope_value_as = {
			name = modeu5_actual_removed_quantity
			value = 0
		}
		save_temporary_scope_value_as = {
			name = modeu5_consistency_validation_required
			value = 1
		}
	}

	save_temporary_scope_value_as = {
		name = modeu5_unsatisfied_quantity
		value = {
			value = scope:modeu5_quantity_requested
			subtract = scope:modeu5_actual_removed_quantity
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_country_stock_after
		value = {
			value = scope:modeu5_country_stock_before
			subtract = scope:modeu5_actual_removed_quantity
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_market_stock_after
		value = {
			value = scope:modeu5_market_stock_before
			subtract = scope:modeu5_actual_removed_quantity
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_available_capacity
		value = {
			value = scope:modeu5_country_stock_cap
			subtract = scope:modeu5_country_stock_before
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_over_capacity_before
		value = {
			value = scope:modeu5_country_stock_before
			subtract = scope:modeu5_country_stock_cap
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_over_capacity_after
		value = {
			value = scope:modeu5_country_stock_after
			subtract = scope:modeu5_country_stock_cap
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_over_capacity_created
		value = 0
	}
	save_temporary_scope_value_as = {
		name = modeu5_stock_difference_after
		value = {
			value = scope:modeu5_market_stock_after
			subtract = scope:modeu5_market_stock_before
			subtract = scope:modeu5_country_stock_after
			add = scope:modeu5_country_stock_before
		}
	}

	if = {
		limit = { scope:modeu5_country_stock_before_raw < 0 }
		error_log = "ModeU5 modeu5_remove_stock found a negative country stock; the country record is normalized and consistency validation is required."
		save_temporary_scope_value_as = {
			name = modeu5_consistency_validation_required
			value = 1
		}
	}
	if = {
		limit = { scope:modeu5_country_stock_cap_raw < 0 }
		error_log = "ModeU5 modeu5_remove_stock found a negative capacity; capacity is treated as zero."
	}

	save_temporary_scope_value_as = {
		name = modeu5_country_record_write_required
		value = 0
	}
	if = {
		limit = {
			OR = {
				scope:modeu5_actual_removed_quantity > 0
				scope:modeu5_country_stock_before_raw < 0
			}
		}
		save_temporary_scope_value_as = {
			name = modeu5_country_record_write_required
			value = 1
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_market_record_write_required
		value = 0
	}
	if = {
		limit = { scope:modeu5_actual_removed_quantity > 0 }
		save_temporary_scope_value_as = {
			name = modeu5_market_record_write_required
			value = 1
		}
	}

}

modeu5_transfer_stock = {
	$seller_country$ = { save_temporary_scope_as = modeu5_seller_country }
	$buyer_country$ = { save_temporary_scope_as = modeu5_buyer_country }
	$source_market$ = { save_temporary_scope_as = modeu5_source_market }
	$target_market$ = { save_temporary_scope_as = modeu5_target_market }
	goods:$good$ = { save_temporary_scope_as = modeu5_good }

	save_temporary_scope_value_as = {
		name = modeu5_quantity_requested_raw
		value = $quantity$
	}
	save_temporary_scope_value_as = {
		name = modeu5_quantity_requested
		value = {
			value = scope:modeu5_quantity_requested_raw
			min = 0
		}
	}
	if = {
		limit = { scope:modeu5_quantity_requested_raw < 0 }
		error_log = "ModeU5 modeu5_transfer_stock received a negative quantity; request normalized to zero."
	}

	modeu5_select_capacity_policy_$target_capacity_policy$ = yes
	modeu5_transfer_stock_good_$good$ = yes
}

modeu5_transfer_stock_calculate = {
	save_temporary_scope_value_as = {
		name = modeu5_seller_stock_before
		value = {
			value = scope:modeu5_seller_stock_before_raw
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_buyer_stock_before
		value = {
			value = scope:modeu5_buyer_stock_before_raw
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_buyer_stock_cap
		value = {
			value = scope:modeu5_buyer_stock_cap_raw
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_available_capacity
		value = {
			value = scope:modeu5_buyer_stock_cap
			subtract = scope:modeu5_buyer_stock_before
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_over_capacity_before
		value = {
			value = scope:modeu5_buyer_stock_before
			subtract = scope:modeu5_buyer_stock_cap
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_same_market_transfer
		value = 0
	}
	if = {
		limit = { scope:modeu5_source_market = scope:modeu5_target_market }
		save_temporary_scope_value_as = {
			name = modeu5_same_market_transfer
			value = 1
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_invalid_same_record_transfer
		value = 0
	}
	if = {
		limit = {
			scope:modeu5_seller_country = scope:modeu5_buyer_country
			scope:modeu5_source_market = scope:modeu5_target_market
		}
		save_temporary_scope_value_as = {
			name = modeu5_invalid_same_record_transfer
			value = 1
		}
	}

	if = {
		limit = {
			scope:modeu5_capacity_policy = 1
			scope:modeu5_invalid_same_record_transfer = 0
		}
		save_temporary_scope_value_as = {
			name = modeu5_actual_transferred_quantity
			value = {
				value = scope:modeu5_quantity_requested
				max = scope:modeu5_seller_stock_before
			}
		}
	}
	else_if = {
		limit = { scope:modeu5_invalid_same_record_transfer = 0 }
		save_temporary_scope_value_as = {
			name = modeu5_actual_transferred_quantity
			value = {
				value = scope:modeu5_quantity_requested
				max = scope:modeu5_seller_stock_before
				max = scope:modeu5_available_capacity
			}
		}
	}
	else = {
		save_temporary_scope_value_as = {
			name = modeu5_actual_transferred_quantity
			value = 0
		}
	}

	save_temporary_scope_value_as = {
		name = modeu5_consistency_validation_required
		value = 0
	}
	# Precision note: keep this strict until TECH-01 113 measures engine and map
	# rounding. A gameplay epsilon would change transfer conservation behavior.
	if = {
		limit = {
			scope:modeu5_same_market_transfer = 0
			OR = {
				scope:modeu5_source_market_stock_before < 0
				scope:modeu5_source_market_stock_before < scope:modeu5_actual_transferred_quantity
			}
		}
		error_log = "ModeU5 modeu5_transfer_stock blocked a source-market aggregate underflow; consistency validation is required."
		save_temporary_scope_value_as = {
			name = modeu5_actual_transferred_quantity
			value = 0
		}
		save_temporary_scope_value_as = {
			name = modeu5_consistency_validation_required
			value = 1
		}
	}

	save_temporary_scope_value_as = {
		name = modeu5_unsatisfied_quantity
		value = {
			value = scope:modeu5_quantity_requested
			subtract = scope:modeu5_actual_transferred_quantity
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_transferred_quantity
		value = scope:modeu5_actual_transferred_quantity
	}
	save_temporary_scope_value_as = {
		name = modeu5_seller_stock_after
		value = {
			value = scope:modeu5_seller_stock_before
			subtract = scope:modeu5_actual_transferred_quantity
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_buyer_stock_after
		value = {
			value = scope:modeu5_buyer_stock_before
			add = scope:modeu5_actual_transferred_quantity
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_source_market_stock_after
		value = scope:modeu5_source_market_stock_before
	}
	save_temporary_scope_value_as = {
		name = modeu5_target_market_stock_after
		value = scope:modeu5_target_market_stock_before
	}
	if = {
		limit = { scope:modeu5_same_market_transfer = 0 }
		save_temporary_scope_value_as = {
			name = modeu5_source_market_stock_after
			value = {
				value = scope:modeu5_source_market_stock_before
				subtract = scope:modeu5_actual_transferred_quantity
			}
		}
		save_temporary_scope_value_as = {
			name = modeu5_target_market_stock_after
			value = {
				value = scope:modeu5_target_market_stock_before
				add = scope:modeu5_actual_transferred_quantity
			}
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_over_capacity_after
		value = {
			value = scope:modeu5_buyer_stock_after
			subtract = scope:modeu5_buyer_stock_cap
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_over_capacity_created
		value = {
			value = scope:modeu5_over_capacity_after
			subtract = scope:modeu5_over_capacity_before
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_stock_difference_after
		value = 0
	}

	if = {
		limit = { scope:modeu5_seller_stock_before_raw < 0 }
		error_log = "ModeU5 modeu5_transfer_stock found a negative seller stock; the seller record is normalized and consistency validation is required."
		save_temporary_scope_value_as = {
			name = modeu5_consistency_validation_required
			value = 1
		}
	}
	if = {
		limit = { scope:modeu5_buyer_stock_before_raw < 0 }
		error_log = "ModeU5 modeu5_transfer_stock found a negative buyer stock; the buyer record is normalized and consistency validation is required."
		save_temporary_scope_value_as = {
			name = modeu5_consistency_validation_required
			value = 1
		}
	}
	if = {
		limit = { scope:modeu5_buyer_stock_cap_raw < 0 }
		error_log = "ModeU5 modeu5_transfer_stock found a negative buyer capacity; capacity is treated as zero."
	}

	save_temporary_scope_value_as = {
		name = modeu5_seller_record_write_required
		value = 0
	}
	if = {
		limit = {
			scope:modeu5_invalid_same_record_transfer = 0
			OR = {
				scope:modeu5_actual_transferred_quantity > 0

modeu5_decay_stock = {
	$country$ = { save_temporary_scope_as = modeu5_country }
	$market$ = { save_temporary_scope_as = modeu5_market }
	goods:$good$ = { save_temporary_scope_as = modeu5_good }

	save_temporary_scope_value_as = {

		value = $decay_rate$
	}
	save_temporary_scope_value_as = {
		name = modeu5_decay_rate
		value = {
			value = scope:modeu5_decay_rate_raw
			min = 0
			max = 1
		}
	}
	if = {
		limit = {
			OR = {
				scope:modeu5_decay_rate_raw < 0
				scope:modeu5_decay_rate_raw > 1
			}
		}
		error_log = "ModeU5 modeu5_decay_stock received an out-of-range decay rate; rate clamped to [0, 1]."
	}

	modeu5_decay_stock_good_$good$ = yes
}

modeu5_decay_stock_calculate = {
	save_temporary_scope_value_as = {
		name = modeu5_country_stock_before
		value = {
			value = scope:modeu5_country_stock_before_raw
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_country_stock_cap
		value = {
			value = scope:modeu5_country_stock_cap_raw
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_decayed_quantity
		value = {
			value = scope:modeu5_country_stock_before
			multiply = scope:modeu5_decay_rate
			max = scope:modeu5_country_stock_before
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_consistency_validation_required
		value = 0
	}

	# Precision note: a fractional decay can expose engine rounding here.
	# Validate/rebuild and inspect the raw operands before weakening this guard.
	if = {
		limit = {
			OR = {
				scope:modeu5_market_stock_before < 0
				scope:modeu5_market_stock_before < scope:modeu5_decayed_quantity
			}
		}
		error_log = "ModeU5 modeu5_decay_stock blocked a market aggregate underflow; consistency validation is required."
		save_temporary_scope_value_as = {
			name = modeu5_decayed_quantity
			value = 0
		}
		save_temporary_scope_value_as = {
			name = modeu5_consistency_validation_required
			value = 1
		}
	}

	save_temporary_scope_value_as = {
		name = modeu5_unsatisfied_quantity
		value = 0
	}
	save_temporary_scope_value_as = {
		name = modeu5_country_stock_after
		value = {
			value = scope:modeu5_country_stock_before
			subtract = scope:modeu5_decayed_quantity
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_market_stock_after
		value = {
			value = scope:modeu5_market_stock_before
			subtract = scope:modeu5_decayed_quantity
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_available_capacity
		value = {
			value = scope:modeu5_country_stock_cap
			subtract = scope:modeu5_country_stock_before
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_over_capacity_before
		value = {
			value = scope:modeu5_country_stock_before
			subtract = scope:modeu5_country_stock_cap
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_over_capacity_after
		value = {
			value = scope:modeu5_country_stock_after
			subtract = scope:modeu5_country_stock_cap
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_over_capacity_created
		value = 0
	}
	save_temporary_scope_value_as = {
		name = modeu5_stock_difference_after
		value = {
			value = scope:modeu5_market_stock_after
			subtract = scope:modeu5_market_stock_before
			subtract = scope:modeu5_country_stock_after
			add = scope:modeu5_country_stock_before
		}
	}

	if = {
		limit = { scope:modeu5_country_stock_before_raw < 0 }
		error_log = "ModeU5 modeu5_decay_stock found a negative country stock; the country record is normalized and consistency validation is required."
		save_temporary_scope_value_as = {
			name = modeu5_consistency_validation_required
			value = 1
		}
	}
	if = {
		limit = { scope:modeu5_country_stock_cap_raw < 0 }
		error_log = "ModeU5 modeu5_decay_stock found a negative capacity; capacity is treated as zero."
	}

	save_temporary_scope_value_as = {
		name = modeu5_country_record_write_required
		value = 0
	}
	if = {
		limit = {
			OR = {
				scope:modeu5_decayed_quantity > 0
				scope:modeu5_country_stock_before_raw < 0
			}
		}
		save_temporary_scope_value_as = {
			name = modeu5_country_record_write_required
			value = 1
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_market_record_write_required
		value = 0
	}
	if = {
		limit = { scope:modeu5_decayed_quantity > 0 }
		save_temporary_scope_value_as = {
			name = modeu5_market_record_write_required
			value = 1
		}
	}

}

modeu5_validate_stock_consistency_prepare = {
	save_temporary_scope_value_as = {
		name = modeu5_validation_expected_market_stock
		value = scope:modeu5_expected_market_stock
	}
	save_temporary_scope_value_as = {
		name = modeu5_validation_actual_market_stock_before
		value = scope:modeu5_scanned_market_stock
	}
	save_temporary_scope_value_as = {
		name = modeu5_validation_difference_before
		value = {
			value = scope:modeu5_validation_actual_market_stock_before
			subtract = scope:modeu5_validation_expected_market_stock
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_validation_absolute_difference
		value = scope:modeu5_validation_difference_before
	}
	if = {
		limit = { scope:modeu5_validation_absolute_difference < 0 }
		save_temporary_scope_value_as = {
			name = modeu5_validation_absolute_difference
			value = {
				value = scope:modeu5_validation_absolute_difference
				multiply = -1
			}
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_validation_inconsistency_detected
		value = 0
	}
	save_temporary_scope_value_as = {
		name = modeu5_validation_inconsistency_severity
		value = 0
	}
	# Exact accounting is intentional. If tiny residuals appear, preserve the
	# signed difference and run the TECH-01 113 precision probe first.
	if = {
		limit = { scope:modeu5_validation_difference_before != 0 }
		save_temporary_scope_value_as = {
			name = modeu5_validation_inconsistency_detected
			value = 1
		}
		save_temporary_scope_value_as = {
			name = modeu5_validation_inconsistency_severity
			value = 1
		}
	}
	if = {
		limit = {
			scope:modeu5_validation_absolute_difference >= modeu5_stock_consistency_prominent_threshold
		}
		save_temporary_scope_value_as = {
			name = modeu5_validation_inconsistency_severity
			value = 2
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_validation_rebuild_called
		value = 0
	}
	save_temporary_scope_value_as = {
		name = modeu5_validation_country_source_count
		value = scope:modeu5_country_source_count
	}
	save_temporary_scope_value_as = {
		name = modeu5_validation_negative_country_source_detected
		value = scope:modeu5_negative_country_source_detected
	}
	save_temporary_scope_value_as = {
		name = modeu5_validation_over_cap_country_source_detected
		value = scope:modeu5_over_cap_country_source_detected
	}
}

modeu5_validate_stock_consistency_finalize = {
	save_temporary_scope_value_as = {
		name = modeu5_validation_actual_market_stock_after
		value = scope:modeu5_scanned_market_stock
	}
	save_temporary_scope_value_as = {
		name = modeu5_validation_difference_after
		value = {
			value = scope:modeu5_validation_actual_market_stock_after
			subtract = scope:modeu5_validation_expected_market_stock
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_validation_failed_after_rebuild
		value = 0
	}
	# A tiny post-write residual may indicate map persistence rounding. Do not
	# hide it with a global epsilon before TECH-01 113 is characterized.
	if = {
		limit = { scope:modeu5_validation_difference_after != 0 }
		save_temporary_scope_value_as = {
			name = modeu5_validation_failed_after_rebuild
			value = 1
		}
	}
}

	if = {
		limit = { modeu5_stock_runtime_ready_trigger = yes }
		set_global_variable = {
			name = modeu5_initialization_last_mode
			value = 2
		}
	}
	else_if = {
		limit = { modeu5_initialization_not_started_trigger = yes }
		set_global_variable = {
			name = modeu5_initialization_state
			value = 1
		}
		modeu5_run_fresh_opening_stock_initialization = yes
		if = {
			limit = { has_global_variable = modeu5_initialization_failure_detected }
			modeu5_fail_initialization = yes
		}
		else = {
			modeu5_commit_successful_initialization = yes
		}
	}
	else_if = {
		limit = { modeu5_initialization_in_progress_trigger = yes }
		modeu5_initialization_mark_failure = { code = 20 }
		modeu5_fail_initialization = yes
	}
	else_if = {
		limit = { modeu5_initialization_failed_trigger = yes }
		set_global_variable = {
			name = modeu5_initialization_last_mode
			value = -1
		}
		debug_log = "ModeU5 CORE-02 initialization is already failed; startup remains diagnostic-only."
	}
	else_if = {
		limit = {
			has_global_variable = modeu5_stock_schema_version
			global_var:modeu5_stock_schema_version = modeu5_current_stock_schema_version
		}
		modeu5_initialization_mark_failure = { code = 21 }
		modeu5_fail_initialization = yes
	}
	else = {
		modeu5_initialization_mark_failure = { code = 22 }
		modeu5_fail_initialization = yes
	}
}

modeu5_prepare_reconciliation_controller = {
	if = {
		limit = { NOT = { exists = scope:modeu5_reconciliation_controller } }
		remove_global_variable = modeu5_reconciliation_controller_selected
		every_country = {
			limit = {
				NOT = { has_global_variable = modeu5_reconciliation_controller_selected }
			}
			save_temporary_scope_as = modeu5_reconciliation_controller
			set_global_variable = modeu5_reconciliation_controller_selected
		}
		remove_global_variable = modeu5_reconciliation_controller_selected
	}
}

modeu5_reconciliation_prepare = {
	set_local_variable = {
		name = modeu5_reconciliation_records_checked
		value = 0
	}
	set_local_variable = {
		name = modeu5_reconciliation_inconsistencies_found
		value = 0
	}
	set_local_variable = {
		name = modeu5_reconciliation_rebuilds_called
		value = 0
	}
	set_local_variable = {
		name = modeu5_reconciliation_failures
		value = 0
	}
}

modeu5_reconciliation_accumulate_last_validation = {
	change_local_variable = {
		name = modeu5_reconciliation_records_checked
		add = 1
	}
	change_local_variable = {
		name = modeu5_reconciliation_inconsistencies_found
		add = scope:modeu5_validation_inconsistency_detected
	}
	change_local_variable = {
		name = modeu5_reconciliation_rebuilds_called
		add = scope:modeu5_validation_rebuild_called
	}
	change_local_variable = {
		name = modeu5_reconciliation_failures
		add = scope:modeu5_validation_failed_after_rebuild
	}
}

modeu5_reconciliation_finalize = {
	save_temporary_scope_value_as = {
		name = modeu5_reconciliation_records_checked_result
		value = local_var:modeu5_reconciliation_records_checked
	}
	save_temporary_scope_value_as = {
		name = modeu5_reconciliation_inconsistencies_found_result
		value = local_var:modeu5_reconciliation_inconsistencies_found
	}
	save_temporary_scope_value_as = {
		name = modeu5_reconciliation_rebuilds_called_result
		value = local_var:modeu5_reconciliation_rebuilds_called
	}
	save_temporary_scope_value_as = {
		name = modeu5_reconciliation_failures_result
		value = local_var:modeu5_reconciliation_failures
	}

	if = {
		limit = {
			modeu5_debug_capture_enabled_trigger = yes
			exists = scope:modeu5_reconciliation_controller
		}
		scope:modeu5_reconciliation_controller = {
			set_variable = {
				name = modeu5_debug_last_reconciliation_type
				value = scope:modeu5_reconciliation_type
			}
			set_variable = {
				name = modeu5_debug_last_reconciliation_records_checked
				value = scope:modeu5_reconciliation_records_checked_result
			}
			set_variable = {
				name = modeu5_debug_last_reconciliation_inconsistencies_found
				value = scope:modeu5_reconciliation_inconsistencies_found_result
			}
			set_variable = {
				name = modeu5_debug_last_reconciliation_rebuilds_called
				value = scope:modeu5_reconciliation_rebuilds_called_result
			}
			set_variable = {
				name = modeu5_debug_last_reconciliation_failures
				value = scope:modeu5_reconciliation_failures_result
			}
			set_variable = {
				name = modeu5_debug_last_reconciliation_cycle_stamp
				value = 0
			}
			set_variable = {
				name = modeu5_debug_last_reconciliation_initialization_gate_passed
				value = 0
			}
		}
	}

	if = {
		limit = { scope:modeu5_reconciliation_failures_result > 0 }
		error_log = "ModeU5 US-11 reconciliation completed with at least one market cache still inconsistent after rebuild."
	}
}

modeu5_run_monthly_stock_cycle = {
	if = {
		limit = { modeu5_stock_runtime_ready_trigger = yes }
		modeu5_run_monthly_capacity_refresh_for_current_country = yes
		modeu5_prepare_monthly_market_seen_registry = yes
		modeu5_prepare_human_relevant_full_ledger_markets = yes
		save_temporary_scope_as = modeu5_country
		modeu5_run_us00_monthly_pipeline_all_goods = yes
		if = {
			limit = { modeu5_audit_enabled_trigger = yes }
			modeu5_run_monthly_stock_reconciliation_once = yes
		}
	}
	else = {
		if = {
			limit = { modeu5_debug_capture_enabled_trigger = yes }
			set_variable = {
				name = modeu5_debug_last_monthly_capacity_refresh_gate_passed
				value = 0
			}
		}
	}
}

modeu5_run_monthly_stock_reconciliation_once = {
	if = {
		limit = {
			modeu5_stock_runtime_ready_trigger = yes
			modeu5_audit_enabled_trigger = yes
		}
		save_temporary_scope_as = modeu5_reconciliation_controller
		save_temporary_scope_value_as = {
			name = modeu5_reconciliation_month_stamp
			value = {
				value = current_year
				multiply = 12
				add = current_month
			}
		}
		if = {
			limit = {
				NOT = { has_global_variable = modeu5_last_monthly_reconciliation_stamp }
			}
			set_global_variable = {
				name = modeu5_last_monthly_reconciliation_stamp
				value = scope:modeu5_reconciliation_month_stamp
			}
			modeu5_run_allowed_stock_consistency_validation = yes
			if = {
				limit = { modeu5_debug_capture_enabled_trigger = yes }
				set_variable = {
					name = modeu5_debug_last_reconciliation_cycle_stamp
					value = scope:modeu5_reconciliation_month_stamp
				}
				set_variable = {
					name = modeu5_debug_last_reconciliation_initialization_gate_passed
					value = 1
				}
			}
		}
		else_if = {
			limit = {
				global_var:modeu5_last_monthly_reconciliation_stamp != scope:modeu5_reconciliation_month_stamp
			}
			set_global_variable = {
				name = modeu5_last_monthly_reconciliation_stamp
				value = scope:modeu5_reconciliation_month_stamp
			}
			modeu5_run_allowed_stock_consistency_validation = yes
			if = {
				limit = { modeu5_debug_capture_enabled_trigger = yes }
				set_variable = {
					name = modeu5_debug_last_reconciliation_cycle_stamp
					value = scope:modeu5_reconciliation_month_stamp
				}
				set_variable = {
					name = modeu5_debug_last_reconciliation_initialization_gate_passed
					value = 1
				}
			}
		}
	}
}

modeu5_run_four_yearly_stock_reconciliation_once = {
	if = {
		limit = { modeu5_stock_runtime_ready_trigger = yes }
		save_temporary_scope_as = modeu5_reconciliation_controller

			name = modeu5_reconciliation_four_year_stamp

		if = {
			limit = {
				NOT = { has_global_variable = modeu5_last_four_yearly_reconciliation_stamp }

				name = modeu5_last_four_yearly_reconciliation_stamp
				value = scope:modeu5_reconciliation_four_year_stamp
			}
			modeu5_validate_active_stock_consistency = yes
			if = {
				limit = { modeu5_debug_capture_enabled_trigger = yes }
				set_variable = {
					name = modeu5_debug_last_reconciliation_cycle_stamp
					value = scope:modeu5_reconciliation_four_year_stamp
				}
				set_variable = {
					name = modeu5_debug_last_reconciliation_initialization_gate_passed
					value = 1
				}
			}
		}
		else_if = {
			limit = {
				global_var:modeu5_last_four_yearly_reconciliation_stamp != scope:modeu5_reconciliation_four_year_stamp
			}
			set_global_variable = {
				name = modeu5_last_four_yearly_reconciliation_stamp
				value = scope:modeu5_reconciliation_four_year_stamp
			}
			modeu5_validate_active_stock_consistency = yes
			if = {
				limit = { modeu5_debug_capture_enabled_trigger = yes }
				set_variable = {
					name = modeu5_debug_last_reconciliation_cycle_stamp
					value = scope:modeu5_reconciliation_four_year_stamp
				}
				set_variable = {
					name = modeu5_debug_last_reconciliation_initialization_gate_passed
					value = 1
				}
			}
		}
	}
}

		name = modeu5_debug_corrupted_market_stock
		value = $value$
	}

	modeu5_debug_corrupt_market_stock_good_$good$ = yes
}
