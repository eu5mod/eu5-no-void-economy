# ModeU5 US-10.0 stock-demand resolver core.
#
# This file provides the common read-only resolver entry point and the shared
# candidate scoring calculation. It intentionally does not call
# modeu5_remove_stock or modeu5_transfer_stock; business mutation remains in
# US-10.1 / US-10.2 callers.
#
# Public API note:
# - `good` remains a caller-supplied resolver parameter.
# - `modeu5_resolve_stock_demand_good_$good$` is not meant to be a separate
#   public API. It is the generated physical storage adapter for the requested
#   good.
# - The generated adapter is still useful because current ModeU5 stock maps are
#   physically split per good, for example `modeu5_<good>_stock_by_market`.
#   A generic resolver can accept `good`, but the map identifiers that read the
#   stock records still need literal per-good generated code.
#
# Implementation status:
# - Candidate construction, hard filters, bucket ordering, and ordered candidate
#   debug output are intentionally left for the generated per-good adapter.
# - This shared file owns only demand normalization, profile loading, common
#   counters, and score calculation.

modeu5_select_stock_demand_type_consumption = {
	save_temporary_scope_value_as = { name = modeu5_demand_type value = 1 }
	save_temporary_scope_value_as = { name = modeu5_us10_demand_type value = 1 }
	modeu5_load_consumption_scoring_profile = yes
}

modeu5_select_stock_demand_type_inter_market_trade = {
	save_temporary_scope_value_as = { name = modeu5_demand_type value = 2 }
	save_temporary_scope_value_as = { name = modeu5_us10_demand_type value = 2 }
	modeu5_load_trade_scoring_profile = yes
}

modeu5_prepare_stock_demand_context = {
	$demanding_country$ = {
		save_temporary_scope_as = modeu5_demanding_country
		save_temporary_scope_as = modeu5_us10_demanding_country
	}
	$market$ = {
		save_temporary_scope_as = modeu5_market
		save_temporary_scope_as = modeu5_us10_market
	}
	goods:$good$ = {
		save_temporary_scope_as = modeu5_good
		save_temporary_scope_as = modeu5_us10_good
	}

	save_temporary_scope_value_as = {
		name = modeu5_requested_quantity_raw
		value = $requested_quantity$
	}
	save_temporary_scope_value_as = {
		name = modeu5_requested_quantity
		value = {
			value = scope:modeu5_requested_quantity_raw
			min = 0
		}
	}
	save_temporary_scope_value_as = { name = modeu5_us10_requested_quantity_raw value = scope:modeu5_requested_quantity_raw }
	save_temporary_scope_value_as = { name = modeu5_us10_requested_quantity value = scope:modeu5_requested_quantity }
	if = {
		limit = { scope:modeu5_requested_quantity_raw < 0 }
		error_log = "ModeU5 US-10 resolver received a negative requested quantity; request normalized to zero."
	}

	modeu5_select_stock_demand_type_$demand_type$ = yes
	modeu5_initialize_stock_demand_resolution_outputs = yes
}

modeu5_resolve_stock_demand = {
	modeu5_prepare_stock_demand_context = {
		demanding_country = $demanding_country$
		market = $market$
		good = $good$
		demand_type = $demand_type$
		requested_quantity = $requested_quantity$
	}

	if = {
		limit = { scope:modeu5_requested_quantity > 0 }
		# Generated implementation hook.
		#
		# The public resolver is good-parametric, but each good adapter must bind
		# literal ModeU5 stock-map identifiers. The generated implementation should:
		# 1. use the market aggregate as a cheap empty-market prefilter;
		# 2. enumerate the already prepared current-market country work cache;
		# 3. apply hard filters before any opinion/trade-advantage read;
		# 4. bucket candidates before attempting any expensive ordering;
		# 5. expose candidate counts, excluded counts/reasons, total available stock,
		#    best score, and enough debug state for US-10-UI/debug.
		modeu5_resolve_stock_demand_good_$good$ = yes
	}
	else = {
		error_log = "ModeU5 US-10 resolver skipped candidate scan because requested quantity is zero."
	}

	modeu5_store_stock_demand_resolution_outputs = yes
}

	save_temporary_scope_value_as = { name = modeu5_us10_candidate_is_own_country value = 0 }
	save_temporary_scope_value_as = { name = modeu5_us10_candidate_is_overlord value = 0 }
	save_temporary_scope_value_as = { name = modeu5_us10_candidate_is_subject value = 0 }
	save_temporary_scope_value_as = { name = modeu5_us10_candidate_is_market_owner value = 0 }
	save_temporary_scope_value_as = { name = modeu5_us10_candidate_buyer_opinion value = 0 }
	save_temporary_scope_value_as = { name = modeu5_us10_candidate_trade_advantage value = 0 }
	save_temporary_scope_value_as = { name = modeu5_us10_candidate_allowed value = 1 }
	save_temporary_scope_value_as = { name = modeu5_us10_candidate_exclusion_reason value = 0 }
	save_temporary_scope_value_as = { name = modeu5_us10_candidate_stock value = 0 }
	save_temporary_scope_value_as = { name = modeu5_us10_candidate_capacity value = 0 }
	save_temporary_scope_value_as = { name = modeu5_us10_candidate_capacity_floor value = 0 }
	save_temporary_scope_value_as = { name = modeu5_us10_candidate_negative_balance_reserve value = 0 }
	save_temporary_scope_value_as = { name = modeu5_us10_candidate_monthly_added value = 0 }
	save_temporary_scope_value_as = { name = modeu5_us10_candidate_monthly_requested value = 0 }
	save_temporary_scope_value_as = { name = modeu5_us10_candidate_monthly_net_balance value = 0 }

	# Bucket ids are intentionally coarse. They provide a cheap deterministic
	# ordering layer before a future full sort is considered.
	# 1 = own stock, 2 = subject/overlord, 3 = market owner, 4 = other foreign.
	save_temporary_scope_value_as = { name = modeu5_us10_candidate_priority_bucket value = 4 }
	save_temporary_scope_value_as = { name = modeu5_us10_candidate_opinion_weight value = 0 }
	save_temporary_scope_value_as = { name = modeu5_us10_candidate_trade_advantage_weight value = 0 }
}

modeu5_mark_stock_candidate_excluded = {
	save_temporary_scope_value_as = { name = modeu5_us10_candidate_allowed value = 0 }
}

modeu5_apply_current_stock_candidate_stock_filters = {
	if = {
		limit = {
			scope:modeu5_us10_candidate_allowed > 0
			NOT = { scope:modeu5_candidate_stock > 0 }
		}
		save_temporary_scope_value_as = { name = modeu5_us10_candidate_exclusion_reason value = 1 }
		modeu5_mark_stock_candidate_excluded = yes
	}
	if = {
		limit = {
			scope:modeu5_us10_candidate_allowed > 0
			NOT = { scope:modeu5_candidate_stock > scope:modeu5_us10_minimum_stock_to_consider }
		}
		save_temporary_scope_value_as = { name = modeu5_us10_candidate_exclusion_reason value = 2 }
		modeu5_mark_stock_candidate_excluded = yes
	}
	if = {
		limit = {
			scope:modeu5_us10_candidate_allowed > 0
			scope:modeu5_candidate_country != scope:modeu5_demanding_country
			scope:modeu5_us10_candidate_capacity_floor > 0
			NOT = { scope:modeu5_candidate_stock > scope:modeu5_us10_candidate_capacity_floor }
		}
		save_temporary_scope_value_as = { name = modeu5_us10_candidate_exclusion_reason value = 3 }
		modeu5_mark_stock_candidate_excluded = yes
	}
	if = {
		limit = {
			scope:modeu5_us10_candidate_allowed > 0
			scope:modeu5_candidate_country != scope:modeu5_demanding_country
			scope:modeu5_us10_candidate_monthly_net_balance < 0
			scope:modeu5_us10_candidate_negative_balance_reserve > 0
			NOT = { scope:modeu5_candidate_stock > scope:modeu5_us10_candidate_negative_balance_reserve }
		}
		save_temporary_scope_value_as = { name = modeu5_us10_candidate_exclusion_reason value = 4 }
		modeu5_mark_stock_candidate_excluded = yes
		if = {
			limit = { exists = scope:modeu5_us10_resolution_controller }
			scope:modeu5_us10_resolution_controller = {
				change_local_variable = { name = modeu5_us10_negative_balance_exclusions add = 1 }
			}
		}
	}
}

modeu5_prepare_current_stock_candidate_relations = {
	if = {
		limit = { scope:modeu5_candidate_country = scope:modeu5_demanding_country }
		save_temporary_scope_value_as = { name = modeu5_us10_candidate_is_own_country value = 1 }
	}
	else = {
		if = {
			limit = {
				scope:modeu5_candidate_country = {
					is_subject_of = scope:modeu5_demanding_country
				}
			}
			save_temporary_scope_value_as = { name = modeu5_us10_candidate_is_subject value = 1 }
		}
		if = {
			limit = {
				scope:modeu5_demanding_country = {
					is_subject_of = scope:modeu5_candidate_country
				}
			}
			save_temporary_scope_value_as = { name = modeu5_us10_candidate_is_overlord value = 1 }
		}
		if = {
			limit = {
				scope:modeu5_market = {
					owner ?= scope:modeu5_candidate_country
				}
			}
			save_temporary_scope_value_as = { name = modeu5_us10_candidate_is_market_owner value = 1 }
		}
	}
}

modeu5_apply_current_stock_candidate_hard_filters = {
	if = {
		limit = {
			scope:modeu5_us10_candidate_allowed > 0
			scope:modeu5_us10_allow_at_war_stocks = 0
			scope:modeu5_candidate_country != scope:modeu5_demanding_country
			scope:modeu5_candidate_country = {
				is_at_war_with = scope:modeu5_demanding_country
			}
		}
		save_temporary_scope_value_as = { name = modeu5_us10_candidate_exclusion_reason value = 5 }
		modeu5_mark_stock_candidate_excluded = yes
	}
	if = {
		limit = {
			scope:modeu5_us10_candidate_allowed > 0
			scope:modeu5_us10_allow_embargoed_stocks = 0
			scope:modeu5_candidate_country != scope:modeu5_demanding_country
			OR = {
				scope:modeu5_candidate_country = {
					is_embargoing = scope:modeu5_demanding_country
				}
				scope:modeu5_candidate_country = {
					is_embargoed_by = scope:modeu5_demanding_country
				}
			}
		}
		save_temporary_scope_value_as = { name = modeu5_us10_candidate_exclusion_reason value = 6 }
		modeu5_mark_stock_candidate_excluded = yes
	}
	if = {
		limit = {
			scope:modeu5_us10_candidate_allowed > 0
			scope:modeu5_us10_allow_subject_stocks = 0
			OR = {
				scope:modeu5_us10_candidate_is_subject > 0
				scope:modeu5_us10_candidate_is_overlord > 0
			}
		}
		save_temporary_scope_value_as = { name = modeu5_us10_candidate_exclusion_reason value = 9 }
		modeu5_mark_stock_candidate_excluded = yes
	}
	if = {
		limit = {
			scope:modeu5_us10_candidate_allowed > 0
			scope:modeu5_us10_allow_market_owner_stock = 0
			scope:modeu5_us10_candidate_is_market_owner > 0
			scope:modeu5_us10_candidate_is_own_country = 0
			scope:modeu5_us10_candidate_is_subject = 0
			scope:modeu5_us10_candidate_is_overlord = 0
		}
		save_temporary_scope_value_as = { name = modeu5_us10_candidate_exclusion_reason value = 10 }
		modeu5_mark_stock_candidate_excluded = yes
	}
	if = {
		limit = {
			scope:modeu5_us10_candidate_allowed > 0
			scope:modeu5_us10_allow_foreign_stocks = 0
			scope:modeu5_us10_candidate_is_own_country = 0
			scope:modeu5_us10_candidate_is_subject = 0
			scope:modeu5_us10_candidate_is_overlord = 0
			scope:modeu5_us10_candidate_is_market_owner = 0
		}
		save_temporary_scope_value_as = { name = modeu5_us10_candidate_exclusion_reason value = 8 }
		modeu5_mark_stock_candidate_excluded = yes
	}
}

modeu5_prepare_current_stock_candidate_priority_bucket = {
	# Cheap pre-ordering. Full score remains available for debug and tie-breaking,
	# but generated candidate scans should avoid an O(n²) generic sort when these
	# buckets are sufficient for the MVP.
	save_temporary_scope_value_as = { name = modeu5_us10_candidate_priority_bucket value = 4 }

	if = {
		limit = { scope:modeu5_us10_candidate_is_own_country > 0 }
		save_temporary_scope_value_as = { name = modeu5_us10_candidate_priority_bucket value = 1 }
	}
	else_if = {
		limit = {
			OR = {
				scope:modeu5_us10_candidate_is_overlord > 0
				scope:modeu5_us10_candidate_is_subject > 0
			}
		}
		save_temporary_scope_value_as = { name = modeu5_us10_candidate_priority_bucket value = 2 }
	}
	else_if = {
		limit = { scope:modeu5_us10_candidate_is_market_owner > 0 }
		save_temporary_scope_value_as = { name = modeu5_us10_candidate_priority_bucket value = 3 }
	}
}

modeu5_calculate_current_stock_candidate_score = {
	# Relation and exposure defaults are prepared by the per-good candidate scan.
	# Future confirmed engine exposure can set the subject / overlord / market-owner
	# flags before this helper runs without changing the scoring formula.
	#
	# Performance rule:
	# Call this only after cheap hard filters have accepted the candidate. Opinion
	# and trade-advantage reads should be prepared by the generated candidate scan
	# only when the matching enable flag and coefficient are non-zero.
	save_temporary_scope_value_as = { name = modeu5_us10_stock_candidate_score value = 0 }
	modeu5_prepare_current_stock_candidate_priority_bucket = yes

	if = {
		limit = { scope:modeu5_us10_candidate_is_own_country > 0 }
		save_temporary_scope_value_as = {
			name = modeu5_us10_stock_candidate_score
			value = {
				value = scope:modeu5_us10_stock_candidate_score
				add = scope:modeu5_us10_own_country_bonus
			}
		}
	}
	if = {
		limit = { scope:modeu5_us10_candidate_is_overlord > 0 }
		save_temporary_scope_value_as = {
			name = modeu5_us10_stock_candidate_score
			value = {
				value = scope:modeu5_us10_stock_candidate_score
				add = scope:modeu5_us10_overlord_bonus
			}
		}
	}
	if = {
		limit = { scope:modeu5_us10_candidate_is_subject > 0 }
		save_temporary_scope_value_as = {
			name = modeu5_us10_stock_candidate_score
			value = {
				value = scope:modeu5_us10_stock_candidate_score
				add = scope:modeu5_us10_subject_bonus
			}
		}
	}
	if = {
		limit = { scope:modeu5_us10_candidate_is_market_owner > 0 }
		save_temporary_scope_value_as = {
			name = modeu5_us10_stock_candidate_score
			value = {
				value = scope:modeu5_us10_stock_candidate_score
				add = scope:modeu5_us10_market_owner_bonus
			}
		}
	}

	if = {
		limit = {
			scope:modeu5_us10_enable_opinion_scoring > 0
			scope:modeu5_us10_opinion_coefficient != 0
		}
		save_temporary_scope_value_as = {
			name = modeu5_us10_candidate_opinion_weight
			value = {
				value = scope:modeu5_us10_candidate_buyer_opinion
				multiply = scope:modeu5_us10_opinion_coefficient
			}
		}
	}
	else = {
		save_temporary_scope_value_as = { name = modeu5_us10_candidate_opinion_weight value = 0 }
	}

	if = {
		limit = {
			scope:modeu5_us10_enable_trade_advantage_scoring > 0
			scope:modeu5_us10_trade_advantage_coefficient != 0
		}
		save_temporary_scope_value_as = {
			name = modeu5_us10_candidate_trade_advantage_weight
			value = {
				value = scope:modeu5_us10_candidate_trade_advantage
				multiply = scope:modeu5_us10_trade_advantage_coefficient
			}
		}
	}
	else = {
		save_temporary_scope_value_as = { name = modeu5_us10_candidate_trade_advantage_weight value = 0 }
	}

	save_temporary_scope_value_as = {
		name = modeu5_us10_stock_candidate_score
		value = {
			value = scope:modeu5_us10_stock_candidate_score
			add = scope:modeu5_us10_candidate_opinion_weight
			add = scope:modeu5_us10_candidate_trade_advantage_weight
		}
	}
}

modeu5_reset_us10_monthly_runtime_counters = {
	set_global_variable = { name = modeu5_us10_monthly_consumption_requests_processed value = 0 }
	set_global_variable = { name = modeu5_us10_monthly_consumption_requested_quantity value = 0 }
	set_global_variable = { name = modeu5_us10_monthly_consumption_satisfied_quantity value = 0 }
	set_global_variable = { name = modeu5_us10_monthly_consumption_unsatisfied_quantity value = 0 }
	set_global_variable = { name = modeu5_us10_monthly_trade_signals_blocked value = 0 }
	set_global_variable = { name = modeu5_us10_monthly_trade_signal_quantity value = 0 }
	set_global_variable = { name = modeu5_us10_monthly_trade_signal_estimated_goods_quantity value = 0 }
}

modeu5_ensure_us10_monthly_runtime_counters = {
	if = { limit = { NOT = { has_global_variable = modeu5_us10_monthly_consumption_requests_processed } } set_global_variable = { name = modeu5_us10_monthly_consumption_requests_processed value = 0 } }
	if = { limit = { NOT = { has_global_variable = modeu5_us10_monthly_consumption_requested_quantity } } set_global_variable = { name = modeu5_us10_monthly_consumption_requested_quantity value = 0 } }
	if = { limit = { NOT = { has_global_variable = modeu5_us10_monthly_consumption_satisfied_quantity } } set_global_variable = { name = modeu5_us10_monthly_consumption_satisfied_quantity value = 0 } }
	if = { limit = { NOT = { has_global_variable = modeu5_us10_monthly_consumption_unsatisfied_quantity } } set_global_variable = { name = modeu5_us10_monthly_consumption_unsatisfied_quantity value = 0 } }
	if = { limit = { NOT = { has_global_variable = modeu5_us10_monthly_trade_signals_blocked } } set_global_variable = { name = modeu5_us10_monthly_trade_signals_blocked value = 0 } }
	if = { limit = { NOT = { has_global_variable = modeu5_us10_monthly_trade_signal_quantity } } set_global_variable = { name = modeu5_us10_monthly_trade_signal_quantity value = 0 } }
	if = { limit = { NOT = { has_global_variable = modeu5_us10_monthly_trade_signal_estimated_goods_quantity } } set_global_variable = { name = modeu5_us10_monthly_trade_signal_estimated_goods_quantity value = 0 } }
}

# US-10.1 / US-10.2 explicit-request runtime wrappers.
#
# These wrappers intentionally delegate requested-quantity normalization and
# demand-type profile loading to the canonical #103 resolver context before
# running the generated mutating adapter. The read-only public resolver still
# performs a candidate scan for diagnostics, but the mutating wrappers do not
# run that scan before their own-stock fast path.

modeu5_resolve_stock_consumption = {
	$consumer_country$ = { save_temporary_scope_as = modeu5_consumer_country }
	$market$ = { save_temporary_scope_as = modeu5_market }
	goods:$good$ = { save_temporary_scope_as = modeu5_good }

	modeu5_prepare_stock_demand_context = {
		demanding_country = scope:modeu5_consumer_country
		market = scope:modeu5_market
		good = $good$
		demand_type = consumption
		requested_quantity = $requested_quantity$
	}

	modeu5_resolve_stock_consumption_good_$good$ = yes
	modeu5_record_country_market_consumption_outcome = {
		country = scope:modeu5_consumer_country
		market = scope:modeu5_market
		good = $good$
		requested_quantity = scope:modeu5_requested_quantity
		satisfied_quantity = scope:modeu5_satisfied_quantity
		unsatisfied_quantity = scope:modeu5_unsatisfied_quantity
	}
}

modeu5_resolve_pop_stock_consumption = {
	$location$ = { save_temporary_scope_as = modeu5_demand_location }
	$consumer_country$ = { save_temporary_scope_as = modeu5_consumer_country }
	$market$ = { save_temporary_scope_as = modeu5_market }
	goods:$good$ = { save_temporary_scope_as = modeu5_good }

	modeu5_resolve_stock_consumption = {
		consumer_country = scope:modeu5_consumer_country
		market = scope:modeu5_market
		good = $good$
		requested_quantity = $requested_quantity$
	}
	modeu5_record_pop_demand_outcome = {
		location = scope:modeu5_demand_location
		good = $good$
		requested_quantity = scope:modeu5_requested_quantity
		satisfied_quantity = scope:modeu5_satisfied_quantity
		unsatisfied_quantity = scope:modeu5_unsatisfied_quantity
	}
}

modeu5_resolve_inter_market_stock_transfer = {
	$buyer_country$ = { save_temporary_scope_as = modeu5_buyer_country }
	$source_market$ = { save_temporary_scope_as = modeu5_source_market }
	$target_market$ = { save_temporary_scope_as = modeu5_target_market }
	goods:$good$ = { save_temporary_scope_as = modeu5_good }

	modeu5_prepare_stock_demand_context = {
		demanding_country = scope:modeu5_buyer_country
		market = scope:modeu5_source_market
		good = $good$
		demand_type = inter_market_trade
		requested_quantity = $requested_quantity$
	}

	if = {
		limit = { scope:modeu5_source_market = scope:modeu5_target_market }
		save_temporary_scope_value_as = { name = modeu5_transferred_quantity value = 0 }
		save_temporary_scope_value_as = { name = modeu5_satisfied_quantity value = 0 }
		save_temporary_scope_value_as = { name = modeu5_unsatisfied_quantity value = scope:modeu5_requested_quantity }
		save_temporary_scope_value_as = { name = modeu5_us10_same_market_transfer_skipped value = 1 }
	}
	else = {
		save_temporary_scope_value_as = { name = modeu5_us10_same_market_transfer_skipped value = 0 }
		modeu5_resolve_inter_market_stock_transfer_good_$good$ = yes
	}
	modeu5_record_country_market_trade_outcome = {
		country = scope:modeu5_buyer_country
		market = scope:modeu5_target_market
		good = $good$
		requested_quantity = scope:modeu5_requested_quantity
		satisfied_quantity = scope:modeu5_transferred_quantity
		unsatisfied_quantity = scope:modeu5_unsatisfied_quantity
	}
}

modeu5_queue_monthly_stock_consumption = {
	$country$ = { save_temporary_scope_as = modeu5_monthly_consumption_country }
	$market$ = { save_temporary_scope_as = modeu5_monthly_consumption_market }
	goods:$good$ = { save_temporary_scope_as = modeu5_good }
	save_temporary_scope_value_as = {
		name = modeu5_monthly_consumption_requested_raw
		value = $requested_quantity$
	}
	save_temporary_scope_value_as = {
		name = modeu5_monthly_consumption_requested_quantity
		value = {
			value = scope:modeu5_monthly_consumption_requested_raw
			min = 0
		}
	}
	if = {
		limit = { scope:modeu5_monthly_consumption_requested_raw < 0 }
		error_log = "ModeU5 monthly stock consumption request received a negative quantity; request normalized to zero."
	}
	if = {
		limit = { scope:modeu5_monthly_consumption_requested_quantity > 0 }
		modeu5_queue_monthly_stock_consumption_good_$good$ = yes
	}
}
