# US-02 storage-capacity calculation.
#
# Capacity is rebuilt from current world state and persisted as one synchronized
# country-scoped map family per good. These effects never mutate stock.

modeu5_recalculate_country_market_capacity = {
	$country$ = { save_temporary_scope_as = modeu5_country }
	$market$ = { save_temporary_scope_as = modeu5_market }
	goods:$good$ = { save_temporary_scope_as = modeu5_good }

	modeu5_recalculate_country_market_capacity_good_$good$ = yes
}

# Reusable CORE-03 numerator. Foreign buildings are excluded because their
# capacity belongs to the building owner, not to the location owner.
modeu5_calculate_location_storage_capacity = {
	$country$ = { save_temporary_scope_as = modeu5_capacity_country }
	$location$ = { save_temporary_scope_as = modeu5_capacity_location }

	scope:modeu5_capacity_country = {
		set_local_variable = { name = modeu5_location_capacity_base_accumulator value = 0 }
		set_local_variable = { name = modeu5_location_capacity_building_accumulator value = 0 }
	}

	scope:modeu5_capacity_location = {
		scope:modeu5_capacity_country = {
			change_local_variable = {
				name = modeu5_location_capacity_base_accumulator
				add = modeu5_storage_capacity_per_owned_location
			}
		}
		every_buildings_in_location = {
			modeu5_calculate_domestic_storage_building_contribution = yes
			scope:modeu5_capacity_country = {
				change_local_variable = {
					name = modeu5_location_capacity_building_accumulator
					add = scope:modeu5_capacity_current_building_contribution
				}
			}
		}
	}

	scope:modeu5_capacity_country = {
		save_temporary_scope_value_as = {
			name = modeu5_location_capacity_base
			value = local_var:modeu5_location_capacity_base_accumulator
		}
		save_temporary_scope_value_as = {
			name = modeu5_location_capacity_building
			value = local_var:modeu5_location_capacity_building_accumulator
		}
		save_temporary_scope_value_as = {
			name = modeu5_location_capacity_total
			value = {
				value = local_var:modeu5_location_capacity_base_accumulator
				add = local_var:modeu5_location_capacity_building_accumulator
			}
		}
	}
}

modeu5_calculate_country_market_capacity = {
	scope:modeu5_country = {
		set_local_variable = { name = modeu5_capacity_base_accumulator value = 0 }
		set_local_variable = { name = modeu5_capacity_building_accumulator value = 0 }
		set_local_variable = { name = modeu5_capacity_foreign_accumulator value = 0 }
		set_local_variable = { name = modeu5_capacity_owned_location_count value = 0 }
		set_local_variable = { name = modeu5_capacity_marketplace_levels value = 0 }
		set_local_variable = { name = modeu5_capacity_market_warehouse_levels value = 0 }
		set_local_variable = { name = modeu5_capacity_funduq_levels value = 0 }
		set_local_variable = { name = modeu5_capacity_caravanserai_levels value = 0 }
		set_local_variable = { name = modeu5_capacity_fondaco_levels value = 0 }

		every_owned_location = {
			limit = { market = scope:modeu5_market }

			scope:modeu5_country = {
				change_local_variable = {
					name = modeu5_capacity_base_accumulator
					add = modeu5_storage_capacity_per_owned_location
				}
				change_local_variable = {
					name = modeu5_capacity_owned_location_count
					add = 1
				}
			}

			every_buildings_in_location = {
				modeu5_calculate_domestic_storage_building_contribution = yes
				scope:modeu5_country = {
					change_local_variable = {
						name = modeu5_capacity_building_accumulator
						add = scope:modeu5_capacity_current_building_contribution
					}
					change_local_variable = {
						name = modeu5_capacity_marketplace_levels
						add = scope:modeu5_capacity_current_marketplace_levels
					}
					change_local_variable = {
						name = modeu5_capacity_market_warehouse_levels
						add = scope:modeu5_capacity_current_market_warehouse_levels
					}
					change_local_variable = {
						name = modeu5_capacity_funduq_levels
						add = scope:modeu5_capacity_current_funduq_levels
					}
					change_local_variable = {
						name = modeu5_capacity_caravanserai_levels
						add = scope:modeu5_capacity_current_caravanserai_levels
					}
				}
			}
		}

		every_owned_foreign_building = {
			limit = {
				building_type = building_type:fondaco
				location = { market = scope:modeu5_market }
			}
			modeu5_calculate_foreign_storage_building_contribution = yes
			scope:modeu5_country = {
				change_local_variable = {
					name = modeu5_capacity_foreign_accumulator
					add = scope:modeu5_capacity_current_building_contribution
				}
				change_local_variable = {
					name = modeu5_capacity_fondaco_levels
					add = scope:modeu5_capacity_current_fondaco_levels
				}
			}
		}

		save_temporary_scope_value_as = {
			name = modeu5_capacity_base
			value = local_var:modeu5_capacity_base_accumulator
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_building
			value = local_var:modeu5_capacity_building_accumulator
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_foreign
			value = local_var:modeu5_capacity_foreign_accumulator
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_total
			value = {
				value = local_var:modeu5_capacity_base_accumulator
				add = local_var:modeu5_capacity_building_accumulator
				add = local_var:modeu5_capacity_foreign_accumulator
			}
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_location_count
			value = local_var:modeu5_capacity_owned_location_count
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_marketplace_level_count
			value = local_var:modeu5_capacity_marketplace_levels
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_market_warehouse_level_count
			value = local_var:modeu5_capacity_market_warehouse_levels
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_funduq_level_count
			value = local_var:modeu5_capacity_funduq_levels
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_caravanserai_level_count
			value = local_var:modeu5_capacity_caravanserai_levels
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_fondaco_level_count
			value = local_var:modeu5_capacity_fondaco_levels
		}
	}
}

modeu5_calculate_domestic_storage_building_contribution = {
	save_temporary_scope_value_as = { name = modeu5_capacity_current_building_contribution value = 0 }
	save_temporary_scope_value_as = { name = modeu5_capacity_current_marketplace_levels value = 0 }
	save_temporary_scope_value_as = { name = modeu5_capacity_current_market_warehouse_levels value = 0 }
	save_temporary_scope_value_as = { name = modeu5_capacity_current_funduq_levels value = 0 }
	save_temporary_scope_value_as = { name = modeu5_capacity_current_caravanserai_levels value = 0 }

	if = {
		limit = { building_type = building_type:marketplace }
		save_temporary_scope_value_as = {
			name = modeu5_capacity_current_building_contribution
			value = {
				value = building_level
				multiply = modeu5_storage_capacity_per_marketplace_level
			}
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_current_marketplace_levels
			value = building_level
		}
	}
	else_if = {
		limit = { building_type = building_type:market_warehouse }
		save_temporary_scope_value_as = {
			name = modeu5_capacity_current_building_contribution
			value = {
				value = building_level
				multiply = modeu5_storage_capacity_per_market_warehouse_level
			}
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_current_market_warehouse_levels
			value = building_level
		}
	}
	else_if = {
		limit = { building_type = building_type:funduq }
		save_temporary_scope_value_as = {
			name = modeu5_capacity_current_building_contribution
			value = {
				value = building_level
				multiply = modeu5_storage_capacity_per_funduq_level
			}
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_current_funduq_levels
			value = building_level
		}
	}
	else_if = {
		limit = { building_type = building_type:caravanserai }
		save_temporary_scope_value_as = {
			name = modeu5_capacity_current_building_contribution
			value = {
				value = building_level
				multiply = modeu5_storage_capacity_per_caravanserai_level
			}
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_current_caravanserai_levels
			value = building_level
		}
	}
}

modeu5_calculate_foreign_storage_building_contribution = {
	save_temporary_scope_value_as = { name = modeu5_capacity_current_building_contribution value = 0 }
	save_temporary_scope_value_as = { name = modeu5_capacity_current_fondaco_levels value = 0 }

	if = {
		limit = { building_type = building_type:fondaco }
		save_temporary_scope_value_as = {
			name = modeu5_capacity_current_building_contribution
			value = {
				value = building_level
				multiply = modeu5_storage_capacity_per_fondaco_level
			}
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_current_fondaco_levels
			value = building_level
		}
	}
}

modeu5_finalize_country_market_capacity = {
	save_temporary_scope_value_as = {
		name = modeu5_capacity_stock
		value = {
			value = scope:modeu5_country_stock_before_raw
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_capacity_available
		value = {
			value = scope:modeu5_capacity_total
			subtract = scope:modeu5_capacity_stock
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_capacity_over_cap
		value = {
			value = scope:modeu5_capacity_stock
			subtract = scope:modeu5_capacity_total
			min = 0
		}
	}
}
