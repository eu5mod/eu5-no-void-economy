# US-02 storage-capacity calculation.
#
# Capacity is rebuilt from current world state and persisted once per
# country-market. The expensive country-owned location scan is performed once
# per country as a pooled location/rank capacity calculation, then divided
# across markets present in that country. Each market keeps its own
# merchant/trade-capacity contribution. Generated per-good stock adapters read
# the shared capacity maps. These effects never mutate stock.

modeu5_recalculate_country_market_capacity = {
	$country$ = { save_temporary_scope_as = modeu5_country }
	$market$ = { save_temporary_scope_as = modeu5_market }
	goods:$good$ = { save_temporary_scope_as = modeu5_good }

	modeu5_recalculate_country_market_capacity_shared = yes
}

modeu5_recalculate_saved_country_market_storage_capacities = {
	modeu5_recalculate_country_market_capacity_shared = yes
}

modeu5_recalculate_saved_country_storage_capacities = {
	scope:modeu5_country = {
		modeu5_calculate_country_storage_capacity_pool = yes
		every_market_present_in_country = {
			save_temporary_scope_as = modeu5_market
			modeu5_recalculate_country_market_capacity_from_prepared_pool_shared = yes
		}
	}
}

modeu5_recalculate_country_storage_capacities = {
	$country$ = { save_temporary_scope_as = modeu5_country }
	scope:modeu5_country = {
		modeu5_rebuild_country_location_capacity_pool = yes
	}
	modeu5_recalculate_saved_country_storage_capacities = yes
}

modeu5_run_monthly_capacity_refresh_for_current_country = {
	save_temporary_scope_as = modeu5_country
	modeu5_recalculate_saved_country_storage_capacities = yes

	save_temporary_scope_value_as = {
		name = modeu5_monthly_capacity_refresh_stamp
		value = {
			value = current_year
			multiply = 12
			add = current_month
		}
	}
	set_variable = {
		name = modeu5_debug_last_monthly_capacity_refresh_stamp
		value = scope:modeu5_monthly_capacity_refresh_stamp
	}
	set_variable = {
		name = modeu5_debug_last_monthly_capacity_refresh_gate_passed
		value = 1
	}
}

modeu5_calculate_location_rank_storage_capacity = {
	save_temporary_scope_value_as = {
		name = modeu5_capacity_current_location_rank_contribution
		value = modeu5_storage_capacity_rural_settlement
	}

	if = {
		limit = { is_capital = yes }
		save_temporary_scope_value_as = {
			name = modeu5_capacity_current_location_rank_contribution
			value = modeu5_storage_capacity_capital
		}
	}
	else_if = {
		limit = { location_rank = location_rank:megalopolis }
		save_temporary_scope_value_as = {
			name = modeu5_capacity_current_location_rank_contribution
			value = modeu5_storage_capacity_megalopolis
		}
	}
	else_if = {
		limit = { location_rank = location_rank:city }
		save_temporary_scope_value_as = {
			name = modeu5_capacity_current_location_rank_contribution
			value = modeu5_storage_capacity_city
		}
	}
	else_if = {
		limit = { location_rank = location_rank:town }
		save_temporary_scope_value_as = {
			name = modeu5_capacity_current_location_rank_contribution
			value = modeu5_storage_capacity_town
		}
	}
}

# Reusable CORE-03 numerator. It captures only the local contribution carried by
# a transferred location. Country-market merchant capacity is recalculated at
# the country-market level and is not duplicated into one location numerator.
modeu5_calculate_location_storage_capacity = {
	$country$ = { save_temporary_scope_as = modeu5_capacity_country }
	$location$ = { save_temporary_scope_as = modeu5_capacity_location }

	scope:modeu5_capacity_country = {
		set_local_variable = { name = modeu5_location_capacity_base_accumulator value = 0 }
	}

	scope:modeu5_capacity_location = {
		modeu5_calculate_location_rank_storage_capacity = yes
		scope:modeu5_capacity_country = {
			change_local_variable = {
				name = modeu5_location_capacity_base_accumulator
				add = scope:modeu5_capacity_current_location_rank_contribution
			}
		}
	}

	scope:modeu5_capacity_country = {
		save_temporary_scope_value_as = {
			name = modeu5_location_capacity_base
			value = local_var:modeu5_location_capacity_base_accumulator
		}
		save_temporary_scope_value_as = {
			name = modeu5_location_capacity_total
			value = local_var:modeu5_location_capacity_base_accumulator
		}
	}
}

modeu5_rebuild_country_location_capacity_pool = {
	scope:modeu5_country = {
		set_local_variable = { name = modeu5_capacity_pool_location_rank_accumulator value = 0 }
		set_local_variable = { name = modeu5_capacity_pool_owned_location_count value = 0 }

		every_owned_location = {
			modeu5_calculate_location_rank_storage_capacity = yes
			scope:modeu5_country = {
				change_local_variable = {
					name = modeu5_capacity_pool_location_rank_accumulator
					add = scope:modeu5_capacity_current_location_rank_contribution
				}
				change_local_variable = {
					name = modeu5_capacity_pool_owned_location_count
					add = 1
				}
			}
		}

		set_variable = {
			name = modeu5_location_capacity_pool_total
			value = local_var:modeu5_capacity_pool_location_rank_accumulator
		}
		set_variable = {
			name = modeu5_location_capacity_pool_location_count
			value = local_var:modeu5_capacity_pool_owned_location_count
		}
	}
}

modeu5_calculate_country_storage_capacity_pool = {
	scope:modeu5_country = {
		set_local_variable = { name = modeu5_capacity_pool_market_count value = 0 }

		every_market_present_in_country = {
			scope:modeu5_country = {
				change_local_variable = { name = modeu5_capacity_pool_market_count add = 1 }
			}
		}

		save_temporary_scope_value_as = { name = modeu5_capacity_pool_location_rank_capacity value = 0 }
		if = {
			limit = { has_variable = modeu5_location_capacity_pool_total }
			save_temporary_scope_value_as = {
				name = modeu5_capacity_pool_location_rank_capacity
				value = var:modeu5_location_capacity_pool_total
			}
		}

		save_temporary_scope_value_as = { name = modeu5_capacity_pool_location_count value = 0 }
		if = {
			limit = { has_variable = modeu5_location_capacity_pool_location_count }
			save_temporary_scope_value_as = {
				name = modeu5_capacity_pool_location_count
				value = var:modeu5_location_capacity_pool_location_count
			}
		}

		save_temporary_scope_value_as = {
			name = modeu5_capacity_pool_market_count
			value = local_var:modeu5_capacity_pool_market_count
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_pool_base
			value = scope:modeu5_capacity_pool_location_rank_capacity
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_pool_total
			value = scope:modeu5_capacity_pool_location_rank_capacity
		}

		save_temporary_scope_value_as = { name = modeu5_capacity_pool_location_rank_per_market value = 0 }

		if = {
			limit = { scope:modeu5_capacity_pool_market_count > 0 }
			save_temporary_scope_value_as = {
				name = modeu5_capacity_pool_location_rank_per_market
				value = {
					value = scope:modeu5_capacity_pool_location_rank_capacity
					divide = scope:modeu5_capacity_pool_market_count
				}
			}
		}
	}
}

modeu5_apply_country_storage_capacity_pool_to_current_market = {
	scope:modeu5_country = {
		save_temporary_scope_value_as = {
			name = modeu5_capacity_current_market_trade_contribution
			value = {
				value = "scope:modeu5_market.merchant_capacity(scope:modeu5_country)"
				multiply = modeu5_storage_capacity_per_trade_capacity
				min = 0
			}
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_base
			value = {
				value = scope:modeu5_capacity_current_market_trade_contribution
				add = scope:modeu5_capacity_pool_location_rank_per_market
			}
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_building
			value = 0
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_foreign
			value = 0
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_trade_capacity
			value = scope:modeu5_capacity_current_market_trade_contribution
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_location_rank_capacity
			value = scope:modeu5_capacity_pool_location_rank_per_market
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_total
			value = {
				value = scope:modeu5_capacity_current_market_trade_contribution
				add = scope:modeu5_capacity_pool_location_rank_per_market
			}
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_location_count
			value = scope:modeu5_capacity_pool_location_count
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_market_count
			value = scope:modeu5_capacity_pool_market_count
		}
		save_temporary_scope_value_as = {
			name = modeu5_capacity_country_pool_total
			value = scope:modeu5_capacity_pool_total
		}
	}
}

modeu5_calculate_country_market_capacity = {
	modeu5_rebuild_country_location_capacity_pool = yes
	modeu5_calculate_country_storage_capacity_pool = yes
	modeu5_apply_country_storage_capacity_pool_to_current_market = yes
}

modeu5_load_capacity_breakdown = {
	scope:modeu5_country = {
		if = {
			limit = {
				has_variable_map = modeu5_stock_cap_by_market
				is_key_in_variable_map = {
					name = modeu5_stock_cap_by_market
					target = scope:modeu5_market
				}
			}
			save_temporary_scope_value_as = {
				name = modeu5_country_stock_cap_raw
				value = "variable_map(modeu5_stock_cap_by_market|scope:modeu5_market)"
			}
		}
		else = {
			save_temporary_scope_value_as = { name = modeu5_country_stock_cap_raw value = 0 }
		}
		if = {
			limit = {
				has_variable_map = modeu5_base_capacity_by_market
				is_key_in_variable_map = {
					name = modeu5_base_capacity_by_market
					target = scope:modeu5_market
				}
			}
			save_temporary_scope_value_as = {
				name = modeu5_country_base_capacity_raw
				value = "variable_map(modeu5_base_capacity_by_market|scope:modeu5_market)"
			}
		}
		else = {
			save_temporary_scope_value_as = { name = modeu5_country_base_capacity_raw value = 0 }
		}
		if = {
			limit = {
				has_variable_map = modeu5_building_capacity_by_market
				is_key_in_variable_map = {
					name = modeu5_building_capacity_by_market
					target = scope:modeu5_market
				}
			}
			save_temporary_scope_value_as = {
				name = modeu5_country_building_capacity_raw
				value = "variable_map(modeu5_building_capacity_by_market|scope:modeu5_market)"
			}
		}
		else = {
			save_temporary_scope_value_as = { name = modeu5_country_building_capacity_raw value = 0 }
		}
		if = {
			limit = {
				has_variable_map = modeu5_foreign_capacity_by_market
				is_key_in_variable_map = {
					name = modeu5_foreign_capacity_by_market
					target = scope:modeu5_market
				}
			}
			save_temporary_scope_value_as = {
				name = modeu5_country_foreign_capacity_raw
				value = "variable_map(modeu5_foreign_capacity_by_market|scope:modeu5_market)"
			}
		}
		else = {
			save_temporary_scope_value_as = { name = modeu5_country_foreign_capacity_raw value = 0 }
		}
	}
}

modeu5_store_capacity_record = {
	scope:modeu5_country = {
		if = {
			limit = {
				has_variable_map = modeu5_stock_cap_by_market
				is_key_in_variable_map = {
					name = modeu5_stock_cap_by_market
					target = scope:modeu5_market
				}
			}
			remove_from_variable_map = {
				name = modeu5_stock_cap_by_market
				key = scope:modeu5_market
			}
		}
		if = {
			limit = { scope:modeu5_capacity_total > 0 }
			add_to_variable_map = {
				name = modeu5_stock_cap_by_market
				key = scope:modeu5_market
				value = scope:modeu5_capacity_total
			}
		}

		if = {
			limit = {
				has_variable_map = modeu5_base_capacity_by_market
				is_key_in_variable_map = {
					name = modeu5_base_capacity_by_market
					target = scope:modeu5_market
				}
			}
			remove_from_variable_map = {
				name = modeu5_base_capacity_by_market
				key = scope:modeu5_market
			}
		}
		if = {
			limit = { scope:modeu5_capacity_base > 0 }
			add_to_variable_map = {
				name = modeu5_base_capacity_by_market
				key = scope:modeu5_market
				value = scope:modeu5_capacity_base
			}
		}

		if = {
			limit = {
				has_variable_map = modeu5_building_capacity_by_market
				is_key_in_variable_map = {
					name = modeu5_building_capacity_by_market
					target = scope:modeu5_market
				}
			}
			remove_from_variable_map = {
				name = modeu5_building_capacity_by_market
				key = scope:modeu5_market
			}
		}
		if = {
			limit = { scope:modeu5_capacity_building > 0 }
			add_to_variable_map = {
				name = modeu5_building_capacity_by_market
				key = scope:modeu5_market
				value = scope:modeu5_capacity_building
			}
		}

		if = {
			limit = {
				has_variable_map = modeu5_foreign_capacity_by_market
				is_key_in_variable_map = {
					name = modeu5_foreign_capacity_by_market
					target = scope:modeu5_market
				}
			}
			remove_from_variable_map = {
				name = modeu5_foreign_capacity_by_market
				key = scope:modeu5_market
			}
		}
		if = {
			limit = { scope:modeu5_capacity_foreign > 0 }
			add_to_variable_map = {
				name = modeu5_foreign_capacity_by_market
				key = scope:modeu5_market
				value = scope:modeu5_capacity_foreign
			}
		}
	}
}

modeu5_prepare_neutral_capacity_stock_context = {
	save_temporary_scope_value_as = { name = modeu5_country_stock_before_raw value = 0 }
}

modeu5_recalculate_country_market_capacity_shared = {
	modeu5_load_capacity_breakdown = yes
	modeu5_prepare_neutral_capacity_stock_context = yes
	modeu5_calculate_country_market_capacity = yes
	modeu5_store_capacity_record = yes
	modeu5_finalize_country_market_capacity = yes
	modeu5_debug_capture_country_market_capacity = yes
}

modeu5_recalculate_country_market_capacity_from_prepared_pool_shared = {
	modeu5_load_capacity_breakdown = yes
	modeu5_prepare_neutral_capacity_stock_context = yes
	modeu5_apply_country_storage_capacity_pool_to_current_market = yes
	modeu5_store_capacity_record = yes
	modeu5_finalize_country_market_capacity = yes
	modeu5_debug_capture_country_market_capacity = yes
}

modeu5_rebuild_and_refresh_country_storage_capacities = {
	save_temporary_scope_as = modeu5_country
	modeu5_rebuild_country_location_capacity_pool = yes
	modeu5_recalculate_saved_country_storage_capacities = yes
}

modeu5_handle_location_capacity_changed = {
	if = {
		limit = { modeu5_stock_runtime_ready_trigger = yes }
		owner ?= {
			modeu5_rebuild_and_refresh_country_storage_capacities = yes
		}
	}
}

modeu5_handle_capital_moved_capacity = {
	if = {
		limit = { modeu5_stock_runtime_ready_trigger = yes }
		modeu5_rebuild_and_refresh_country_storage_capacities = yes
	}
}

modeu5_finalize_country_market_capacity = {
	save_temporary_scope_value_as = {
		name = modeu5_capacity_stock
		value = {
			value = scope:modeu5_country_stock_before_raw
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_capacity_available
		value = {
			value = scope:modeu5_capacity_total
			subtract = scope:modeu5_capacity_stock
			min = 0
		}
	}
	save_temporary_scope_value_as = {
		name = modeu5_capacity_over_cap
		value = {
			value = scope:modeu5_capacity_stock
			subtract = scope:modeu5_capacity_total
			min = 0
		}
	}
}
