modeu5_debug_replace_country_wheat_capacity = {
	$country$ = {
		if = {
			limit = {
				has_variable_map = modeu5_wheat_stock_cap_by_market
				is_key_in_variable_map = {
					name = modeu5_wheat_stock_cap_by_market
					target = $market$
				}
			}
			remove_from_variable_map = {
				name = modeu5_wheat_stock_cap_by_market
				key = $market$
			}
		}
		add_to_variable_map = {
			name = modeu5_wheat_stock_cap_by_market
			key = $market$
			value = $value$
		}
	}
}

modeu5_read_country_stock_good_wheat = {
	$market$ = { save_temporary_scope_as = modeu5_test_record_market }
	$country$ = {
		if = {
			limit = {
				has_variable_map = modeu5_wheat_stock_by_market
				is_key_in_variable_map = {
					name = modeu5_wheat_stock_by_market
					target = scope:modeu5_test_record_market
				}
			}
			save_temporary_scope_value_as = {
				name = $output$
				value = "variable_map(modeu5_wheat_stock_by_market|scope:modeu5_test_record_market)"
			}
		}
		else = {
			save_temporary_scope_value_as = {
				name = $output$
				value = 0
			}
		}
	}
}
