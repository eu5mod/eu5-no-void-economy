# ModeU5 sparse traversal helpers and debug-only performance counters.
#
# PERF-02 builds the country -> native market side. PERF-03 adds a
# market -> countries-present work cache in modeu5_market_country_cache_effects.
# PERF-13 counters are intentionally inert in normal runtime unless a
# debug/test path calls them. They do not mutate stock.

modeu5_clear_performance_relevant_markets = {
	remove_global_variable = modeu5_performance_human_country_count
	remove_global_variable = modeu5_performance_relevant_market_native_iterations
	remove_global_variable = modeu5_performance_relevant_market_added_count

	set_global_variable = { name = modeu5_performance_human_country_count value = 0 }
	set_global_variable = { name = modeu5_performance_relevant_market_native_iterations value = 0 }
	set_global_variable = { name = modeu5_performance_relevant_market_added_count value = 0 }

	if = {
		limit = { has_global_variable_list = modeu5_performance_relevant_markets }
		clear_global_variable_list = modeu5_performance_relevant_markets
	}
}

modeu5_prepare_performance_relevant_market_counters = {
	if = {
		limit = { NOT = { has_global_variable = modeu5_performance_human_country_count } }
		set_global_variable = { name = modeu5_performance_human_country_count value = 0 }
	}
	if = {
		limit = { NOT = { has_global_variable = modeu5_performance_relevant_market_native_iterations } }
		set_global_variable = { name = modeu5_performance_relevant_market_native_iterations value = 0 }
	}
	if = {
		limit = { NOT = { has_global_variable = modeu5_performance_relevant_market_added_count } }
		set_global_variable = { name = modeu5_performance_relevant_market_added_count value = 0 }
	}
}

modeu5_mark_performance_relevant_market = {
	modeu5_prepare_performance_relevant_market_counters = yes

	if = {
		limit = {
			NOT = { has_global_variable_list = modeu5_performance_relevant_markets }
		}
		add_to_global_variable_list = {
			name = modeu5_performance_relevant_markets
			target = scope:modeu5_performance_relevant_market
		}
		set_global_variable = {
			name = modeu5_performance_relevant_market_added_count
			value = {
				value = global_var:modeu5_performance_relevant_market_added_count
				add = 1
			}
		}
	}
	else_if = {
		limit = {
			NOT = {
				is_target_in_global_variable_list = {
					name = modeu5_performance_relevant_markets
					target = scope:modeu5_performance_relevant_market
				}
			}
		}
		add_to_global_variable_list = {
			name = modeu5_performance_relevant_markets
			target = scope:modeu5_performance_relevant_market
		}
		set_global_variable = {
			name = modeu5_performance_relevant_market_added_count
			value = {
				value = global_var:modeu5_performance_relevant_market_added_count
				add = 1
			}
		}
	}
}

modeu5_mark_human_relevant_market_for_country = {
	$country$ = { save_temporary_scope_as = modeu5_performance_country }
	$market$ = { save_temporary_scope_as = modeu5_performance_relevant_market }

	if = {
		limit = {
			modeu5_performance_mode_enabled_trigger = yes
			scope:modeu5_performance_country = { is_ai = no }
		}
		modeu5_mark_performance_relevant_market = yes
	}
}

modeu5_mark_human_relevant_market_if_present = {
	$country$ = { save_temporary_scope_as = modeu5_performance_country }
	$market$ = {
		save_temporary_scope_as = modeu5_performance_target_market
		save_temporary_scope_as = modeu5_performance_relevant_market
	}

	save_temporary_scope_value_as = { name = modeu5_performance_target_market_present value = 0 }

	if = {
		limit = {
			modeu5_performance_mode_enabled_trigger = yes
			scope:modeu5_performance_country = { is_ai = no }
		}
		scope:modeu5_performance_country = {
			set_local_variable = { name = modeu5_performance_target_market_present value = 0 }
			every_market_present_in_country = {
				save_temporary_scope_as = modeu5_performance_iterated_market
				if = {
					limit = {
						scope:modeu5_performance_iterated_market = scope:modeu5_performance_target_market
					}
					scope:modeu5_performance_country = {
						set_local_variable = { name = modeu5_performance_target_market_present value = 1 }
					}
				}
			}
			save_temporary_scope_value_as = {
				name = modeu5_performance_target_market_present
				value = local_var:modeu5_performance_target_market_present
			}
		}
		if = {
			limit = { scope:modeu5_performance_target_market_present > 0 }
			modeu5_mark_performance_relevant_market = yes
		}
	}
}

modeu5_add_country_present_markets_to_performance_relevant_list = {
	modeu5_prepare_performance_relevant_market_counters = yes

	scope:modeu5_performance_country = {
		every_market_present_in_country = {
			save_temporary_scope_as = modeu5_performance_relevant_market
			set_global_variable = {
				name = modeu5_performance_relevant_market_native_iterations
				value = {
					value = global_var:modeu5_performance_relevant_market_native_iterations
					add = 1
				}
			}
			modeu5_mark_performance_relevant_market = yes
		}
	}
}

modeu5_mark_human_relevant_markets_for_country = {
	$country$ = { save_temporary_scope_as = modeu5_performance_country }

	if = {
		limit = {
			modeu5_performance_mode_enabled_trigger = yes
			scope:modeu5_performance_country = { is_ai = no }
		}
		modeu5_add_country_present_markets_to_performance_relevant_list = yes
	}
}

modeu5_rebuild_current_country_relevant_markets = {
	save_temporary_scope_as = modeu5_performance_country
	modeu5_clear_performance_relevant_markets = yes
	modeu5_add_country_present_markets_to_performance_relevant_list = yes
}

modeu5_rebuild_performance_relevant_markets_for_country = {
	$country$ = { save_temporary_scope_as = modeu5_performance_country }
	modeu5_clear_performance_relevant_markets = yes
	modeu5_add_country_present_markets_to_performance_relevant_list = yes
}

modeu5_rebuild_human_relevant_markets = {
	modeu5_clear_performance_relevant_markets = yes

	every_country = {
		limit = { is_ai = no }
		save_temporary_scope_as = modeu5_performance_country
		set_global_variable = {
			name = modeu5_performance_human_country_count
			value = {
				value = global_var:modeu5_performance_human_country_count
				add = 1
			}
		}
		modeu5_add_country_present_markets_to_performance_relevant_list = yes
	}
}

modeu5_prepare_human_relevant_full_ledger_markets = {
	if = {
		limit = { modeu5_accounting_human_relevant_persistence_trigger = yes }
		save_temporary_scope_value_as = {
			name = modeu5_human_relevant_full_ledger_stamp
			value = {
				value = current_year
				multiply = 12
				add = current_month
			}
		}
		if = {
			limit = {
				OR = {
					NOT = { has_global_variable = modeu5_human_relevant_full_ledger_stamp }
					global_var:modeu5_human_relevant_full_ledger_stamp != scope:modeu5_human_relevant_full_ledger_stamp
				}
			}
			set_global_variable = {
				name = modeu5_human_relevant_full_ledger_stamp
				value = scope:modeu5_human_relevant_full_ledger_stamp
			}
			modeu5_rebuild_human_relevant_markets = yes
		}
	}
}

	if = {
		limit = {
			NOT = { has_global_variable_list = modeu5_detailed_accounting_promoted_markets }
		}
		add_to_global_variable_list = {
			name = modeu5_detailed_accounting_promoted_markets
			target = scope:modeu5_promotion_market
		}
	}
	else_if = {
		limit = {
			NOT = {
				is_target_in_global_variable_list = {
					name = modeu5_detailed_accounting_promoted_markets
					target = scope:modeu5_promotion_market
				}
			}
		}
		add_to_global_variable_list = {
			name = modeu5_detailed_accounting_promoted_markets
			target = scope:modeu5_promotion_market
		}
	}
}

modeu5_refresh_market_promotion_country_capacities = {
	scope:modeu5_promotion_market = { save_temporary_scope_as = modeu5_market_country_cache_market }
	modeu5_rebuild_countries_present_in_market = yes

	if = {
		limit = { has_global_variable_list = modeu5_countries_present_in_market }
		every_in_global_list = {
			variable = modeu5_countries_present_in_market
			save_temporary_scope_as = modeu5_country
			scope:modeu5_promotion_market = { save_temporary_scope_as = modeu5_market }
			modeu5_recalculate_country_market_capacity_shared = yes
			set_global_variable = {
				name = modeu5_perf14_promotion_capacity_country_count
				value = {
					value = global_var:modeu5_perf14_promotion_capacity_country_count
					add = 1
				}
			}
		}
	}
}

modeu5_promote_market_to_detailed_accounting = {
	$market$ = {
		save_temporary_scope_as = modeu5_promotion_market
		save_temporary_scope_as = modeu5_market
	}

	modeu5_reset_market_promotion_metrics = yes

	if = {
		limit = { modeu5_market_detailed_accounting_promoted_trigger = yes }
		set_global_variable = { name = modeu5_perf14_promotion_last_result value = 1 }
		set_global_variable = { name = modeu5_perf14_promotion_noop_already_promoted value = 1 }
	}
	else_if = {
		limit = {
			OR = {
				NOT = { has_global_variable_list = modeu5_performance_relevant_markets }
				NOT = {
					is_target_in_global_variable_list = {
						name = modeu5_performance_relevant_markets
						target = scope:modeu5_promotion_market
					}
				}
			}
		}
		set_global_variable = { name = modeu5_perf14_promotion_blocked_not_human_relevant value = 1 }
	}
	else = {
		modeu5_refresh_market_promotion_country_capacities = yes
		modeu5_promote_market_to_detailed_accounting_all_goods = yes

		if = {
			limit = {
				OR = {
					NOT = { has_global_variable = modeu5_perf14_promotion_failure_detected }
					global_var:modeu5_perf14_promotion_failure_detected = 0
				}
			}
			modeu5_mark_market_detailed_accounting_promoted = yes
			set_global_variable = { name = modeu5_perf14_promotion_last_result value = 1 }
		}
	}
}
